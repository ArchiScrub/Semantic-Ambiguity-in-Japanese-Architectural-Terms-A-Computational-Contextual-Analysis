def _make_logger() -> logging.Logger:
    """Console+file logger for the Paper 4 pipeline."""
    logger = logging.getLogger("paper4")
    if logger.handlers:
        return logger
    logger.setLevel(logging.INFO)
    fmt = logging.Formatter("[%(asctime)s] %(levelname)s | %(message)s", "%H:%M:%S")

    ch = logging.StreamHandler()
    ch.setFormatter(fmt)
    ch.setLevel(logging.INFO)

    fh = logging.FileHandler(LOGDIR / "paper4.log", encoding="utf-8")
    fh.setFormatter(fmt)
    fh.setLevel(logging.INFO)

    logger.addHandler(ch)
    logger.addHandler(fh)
    return logger

class Timer:
    """Wall-clock timer for logging stage durations."""

    def __init__(self, label: str) -> None:
        self.label = label
        self.t0: float = 0.0

    def __enter__(self) -> "Timer":
        self.t0 = time.perf_counter()
        LOGGER.info("%s ...", self.label)
        return self

    def __exit__(self, exc_type, exc, tb) -> None:
        dt = time.perf_counter() - self.t0
        LOGGER.info("%s completed in %.3fs", self.label, dt)


def sha256_file(path: Path) -> str:
    """Return short SHA-256 hex digest (12 chars) for a file path, or 'NA' if missing."""
    if not path or not path.exists() or not path.is_file():
        return "NA"
    hasher = hashlib.sha256()
    with open(path, "rb") as file_obj:
        for chunk in iter(lambda: file_obj.read(1 << 20), b""):
            hasher.update(chunk)
    return hasher.hexdigest()[:12]

def cosine(vec_a: NDArray[np.float_], vec_b: NDArray[np.float_]) -> float:
    """Compute cosine similarity between two vectors, returning NaN if a norm is zero."""
    norm_a = float(np.linalg.norm(vec_a))
    norm_b = float(np.linalg.norm(vec_b))
    if norm_a == 0.0 or norm_b == 0.0:
        return float("nan")
    dot_val = float(np.dot(vec_a, vec_b))
    return dot_val / (norm_a * norm_b)

def scaled_figsize(
    n_cols: int,
    n_rows: int,
    min_w: float,
    min_h: float,
    col_scale: float,
    row_scale: float,
) -> Tuple[float, float]:
    """Return a float figsize tuple scaled to table dimensions."""
    fig_w = max(min_w, float(n_cols) * col_scale)
    fig_h = max(min_h, float(n_rows) * row_scale)
    return fig_w, fig_h

def normalize_for_pipeline(text: str) -> str:
    """Apply Unicode NFC normalization and configured replacement mapping."""
    text = unicodedata.normalize("NFC", text)

    # Longest keys first avoids partial replacement problems.
    for key, value in sorted(
        NORMALISATION_MAP.items(),
        key=lambda kv: len(str(kv[0])),
        reverse=True,
    ):
        key_n = unicodedata.normalize("NFC", str(key))
        value_n = unicodedata.normalize("NFC", str(value))
        text = text.replace(key_n, value_n)

    return text


def read_sentences(corpus_path: Path) -> List[str]:
    """Read corpus text, normalize, and split into non-empty sentence units."""
    text = corpus_path.read_text(encoding="utf-8")
    text = normalize_for_pipeline(text)
    sentences = [sent.strip() for sent in SENT_SPLIT_RE.split(text) if sent.strip()]
    return sentences

def sentence_hash(text: str) -> str:
    """Stable sentence hash after normalization/splitting."""
    return hashlib.sha256(text.encode("utf-8")).hexdigest()


def audit_clean_subset_relation() -> None:
    """Check whether CLEAN is an exact sentence-level subset of FULL."""
    full_sentences = read_sentences(resolve_bert_path("FULL"))
    clean_sentences = read_sentences(resolve_bert_path("CLEAN"))

    full_hashes = {sentence_hash(sent) for sent in full_sentences}
    clean_hashes = {sentence_hash(sent) for sent in clean_sentences}
    missing_from_full = clean_hashes - full_hashes

    audit_payload = {
        "full_sentence_count": int(len(full_sentences)),
        "clean_sentence_count": int(len(clean_sentences)),
        "clean_sentences_not_in_full": int(len(missing_from_full)),
        "clean_subset_by_exact_sentence_hash": bool(len(missing_from_full) == 0),
    }

    out_path = RES / "clean_subset_audit.json"
    out_path.write_text(
        json.dumps(audit_payload, ensure_ascii=False, indent=2),
        encoding="utf-8",
    )

    if missing_from_full:
        LOGGER.warning(
            "CLEAN is not an exact sentence-level subset of FULL after normalization/splitting: %d CLEAN sentences not found in FULL.",
            len(missing_from_full),
        )
    else:
        LOGGER.info("CLEAN passed sentence-level subset audit against FULL.")

    LOGGER.info(
        "Torch device: %s | cuda.is_available=%s | cudnn.enabled=%s",
        device,

    )

    banner("P4 RUNTIME")
    LOGGER.info("TARGETS (%d): %s", len(TARGETS), ", ".join(TARGETS))
    LOGGER.info("CATEGORY LABELS (%d): %s", len(CATEGORY_LABELS), ", ".join(CATEGORY_LABELS))

def fdr_bh(p_vals: List[float]) -> List[float]:
    """Benjamini–Hochberg FDR correction."""
    p = np.asarray(p_vals, dtype=float)
    n = p.size
    if n == 0:
        return []
    order = np.argsort(p)
    p_sorted = p[order]
    ranks = np.arange(1, n + 1, dtype=float)
    q_sorted = (p_sorted * n) / ranks
    q_sorted = np.minimum.accumulate(q_sorted[::-1])[::-1]
    q = np.empty_like(q_sorted)
    q[order] = q_sorted
    return np.clip(q, 0.0, 1.0).tolist()

def holm_bonferroni(p_vals: List[float]) -> List[float]:
    """Holm–Bonferroni step-down adjusted p-values."""
    p = np.asarray(p_vals, dtype=float)
    n = int(p.size)
    if n == 0:
        return []
    order = np.argsort(p)
    p_sorted = p[order]
    factors = (n - np.arange(n)).astype(float)
    adj_sorted = np.minimum(p_sorted * factors, 1.0)
    adj_sorted = np.maximum.accumulate(adj_sorted)
    adj = np.empty_like(adj_sorted)
    adj[order] = adj_sorted
    return adj.tolist()

def load_w2v(mode: str, flavour: str) -> Optional[Word2Vec]:
    """Load a cached Word2Vec model (cbow or sg) for a given mode, if available."""
    # Expect Word2Vec models saved by Paper 1 at: models/<MODE>/w2v_cbow.model / w2v_sg.model
    root = cache_root_for(mode)

    model_path = root / name
    if model_path.exists():
        try:

        except (OSError, ValueError) as exc:

    return None

def load_bert_finetuned(mode: str) -> Optional[Tuple[Any, BertTokenizer, int, Path]]:
    """Load the best available fine-tuned BERT masked LM and tokenizer for a mode.

    Resolution order:
    1. adapter root (adapter_config.json + adapter_model.safetensors)
    2. trainer_state.json -> best_model_checkpoint
    3. single valid checkpoint-* directory
    4. final/ directory
    """
    root = cache_root_for(mode)
    bert_root = root / "bert_cased"

    if not bert_root.is_dir():
        LOGGER.warning("Fine-tuned BERT root not found for %s: %s", mode, bert_root)
        return None

    def _has_model_files(p: Path) -> bool:
        return (p / "config.json").is_file() and (
            (p / "pytorch_model.bin").is_file() or (p / "model.safetensors").is_file()
        )

    def _epoch_from_name(p: Path) -> int:
        match = re.search(r"checkpoint-(\d+)$", p.name)
        return int(match.group(1)) if match else -1

    # 1) adapter root
    adapter_cfg = bert_root / "adapter_config.json"
    adapter_weights = bert_root / "adapter_model.safetensors"
    if adapter_cfg.is_file() and adapter_weights.is_file():
        try:
            from peft import PeftConfig, PeftModel

            peft_cfg = PeftConfig.from_pretrained(str(bert_root))
            base_name = str(peft_cfg.base_model_name_or_path)

            try:
                tokenizer = BertTokenizer.from_pretrained(str(bert_root))
            except (OSError, ValueError):
                tokenizer = BertTokenizer.from_pretrained(base_name)

            base_model = BertForMaskedLM.from_pretrained(base_name).to(device)
            model = PeftModel.from_pretrained(base_model, str(bert_root)).to(device)

            LOGGER.info("Loaded fine-tuned BERT for %s from adapter root: %s", mode, bert_root)
            return model, tokenizer, -1, bert_root
        except Exception as exc:
            LOGGER.warning("Adapter load failed for %s at %s: %s", mode, bert_root, exc)

    # 2) trainer_state.json -> best checkpoint
    trainer_state = bert_root / "trainer_state.json"
    if trainer_state.is_file():
        try:
            state = json.loads(trainer_state.read_text(encoding="utf-8"))
            best_rel = str(state.get("best_model_checkpoint", "")).strip()
        except json.JSONDecodeError as exc:
            LOGGER.warning("Could not parse %s: %s", trainer_state, exc)
            best_rel = ""

        if best_rel:
            best_path = Path(best_rel)
            ckpt = best_path if best_path.is_absolute() else (bert_root / best_rel)
            if ckpt.is_dir() and _has_model_files(ckpt):
                tokenizer = BertTokenizer.from_pretrained(str(ckpt))
                model = BertForMaskedLM.from_pretrained(str(ckpt)).to(device)
                best_epoch = _epoch_from_name(ckpt)
                LOGGER.info("Loaded fine-tuned BERT for %s from trainer_state.json: %s", mode, ckpt)
                return model, tokenizer, best_epoch, ckpt
            LOGGER.warning("trainer_state.json pointed to invalid checkpoint for %s: %s", mode, ckpt)

    # 3) single valid checkpoint-* directory
    ckpts = sorted(
        [p for p in bert_root.glob("checkpoint-*") if p.is_dir() and _has_model_files(p)],
        key=_epoch_from_name,
    )
    if len(ckpts) == 1:
        ckpt = ckpts[0]
        tokenizer = BertTokenizer.from_pretrained(str(ckpt))
        model = BertForMaskedLM.from_pretrained(str(ckpt)).to(device)
        best_epoch = _epoch_from_name(ckpt)
        LOGGER.info("Loaded fine-tuned BERT for %s from sole checkpoint dir: %s", mode, ckpt)
        return model, tokenizer, best_epoch, ckpt

    # 4) final/ directory
    final_dir = bert_root / "final"
    if final_dir.is_dir() and _has_model_files(final_dir):
        tokenizer = BertTokenizer.from_pretrained(str(final_dir))
        model = BertForMaskedLM.from_pretrained(str(final_dir)).to(device)
        LOGGER.info("Loaded fine-tuned BERT for %s from final/: %s", mode, final_dir)
        return model, tokenizer, -1, final_dir

    LOGGER.warning("No valid fine-tuned BERT checkpoint resolved for %s under %s", mode, bert_root)
    return None

@torch.no_grad()
def bert_vec(
    model: Any,
    tokenizer: BertTokenizer,
    text: str,
    is_sentence: bool,
) -> Optional[np.ndarray]:
    """Return a BERT vector for a sentence or word/phrase using the last hidden layer."""
    if is_sentence:
        inputs = tokenizer(
            text,
            return_tensors="pt",
            truncation=True,
            max_length=128,
        ).to(device)
        output = model(**inputs, output_hidden_states=True)
        return (
            output.hidden_states[-1]
            .mean(dim=1)
            .squeeze(0)
            .detach()
            .cpu()
            .numpy()
        )

    # word/phrase: use tokens (subword-mean), excluding CLS/SEP
    inputs = tokenizer(
        text,
        return_tensors="pt",
        truncation=True,
        max_length=16,
    ).to(device)
    output = model(**inputs, output_hidden_states=True)
    hidden_states = output.hidden_states[-1][0]  # [seq_len, hidden]
    if hidden_states.shape[0] <= 2:
        return hidden_states.mean(dim=0).cpu().numpy()
    return hidden_states[1:-1].mean(dim=0).cpu().numpy()

def w2v_vec(model: Word2Vec, text: str) -> Optional[np.ndarray]:
    """Return mean Word2Vec vector for tokens present in the model vocabulary, or None if none match."""
    tokens = text.lower().split()
    vectors = [model.wv[token] for token in tokens if token in model.wv]
    if not vectors:
        return None
    return np.mean(np.stack(vectors), axis=0)

def term_pattern(term: str):
    """Compile a Unicode-aware target-term pattern."""
    term_n = unicodedata.normalize("NFC", term)

    # Explicit shōji variants because this term exposed the extraction problem.
    if term_n == "shōji":
        return re_rx.compile(
            r"(?<![\p{L}\p{N}_])(?:shōji|sho\u0304ji|shoji|shouji|障子|しょうじ)(?![\p{L}\p{N}_])",
            re_rx.IGNORECASE,
        )

    # Avoid false matches such as al-Mu'tasim while keeping standalone mu,
    # Japanese-script variants, and hyphenated philosophical forms such as mu-koan.
    if term_n == "mu":
        return re_rx.compile(
            r"(?<![\p{L}\p{N}_-])(?:mu|無|む)(?![\p{L}\p{N}_'’])",
            re_rx.IGNORECASE,
        )

    return re_rx.compile(
        rf"(?<![\p{{L}}\p{{N}}_]){re_rx.escape(term_n)}(?![\p{{L}}\p{{N}}_])",
        re_rx.IGNORECASE,
    )


def find_contexts_with_spans(term: str, sentences: List[str]) -> List[Tuple[int, int, int]]:
    """Return sentence index plus character span for each target occurrence."""
    pattern = term_pattern(term)
    hits: List[Tuple[int, int, int]] = []

    for sent_idx, sent in enumerate(sentences):
        sent_n = unicodedata.normalize("NFC", sent)
        for match in pattern.finditer(sent_n):
            hits.append((sent_idx, match.start(), match.end()))

    return hits


def find_contexts(term: str, sentences: List[str]) -> List[int]:
    """Backward-compatible sentence-index finder."""
    return sorted({sent_idx for sent_idx, _, _ in find_contexts_with_spans(term, sentences)})


def crop_around_target(
    sentence: str,
    char_start: int,
    char_end: int,
    window_chars: int = CONTEXT_WINDOW_CHARS,
) -> Tuple[str, int, int]:
    """Crop around the target so the target cannot be lost by BERT truncation."""
    half_window = max(100, int(window_chars) // 2)

    left = max(0, char_start - half_window)
    right = min(len(sentence), char_end + half_window)

    cropped = sentence[left:right]
    new_start = char_start - left
    new_end = char_end - left

    return cropped, new_start, new_end


@torch.no_grad()
def contextual_embeddings_for(
    term: str,
    sentences: List[str],
    model: Any,
    tokenizer: BertTokenizer,
    allow_sentence_fallback: bool = False,
) -> Tuple[np.ndarray, List[int], Dict[str, int]]:
    """
    Return one contextual embedding per matched sentence.

    If a sentence contains multiple occurrences of the target term, each occurrence
    is extracted with offset mapping and then averaged into one sentence-level
    target vector. This keeps the analysis aligned with the manuscript's
    sentence-context framing.
    """
    hits = find_contexts_with_spans(term, sentences)

    hits_by_sentence: Dict[int, List[Tuple[int, int]]] = {}
    for sent_idx, char_start, char_end in hits:
        hits_by_sentence.setdefault(sent_idx, []).append((char_start, char_end))

    embeddings: List[np.ndarray] = []
    kept_indices: List[int] = []

    audit = {
        "candidate_occurrences": int(len(hits)),
        "candidate_sentences": int(len(hits_by_sentence)),
        "matched_span_occurrences": 0,
        "matched_span_vectors": 0,
        "sentence_mean_fallback_vectors": 0,
        "span_fail_truncated_or_no_overlap": 0,
    }

    for sent_idx, spans in hits_by_sentence.items():
        original_sentence = unicodedata.normalize("NFC", sentences[sent_idx])
        occurrence_vectors: List[np.ndarray] = []

        for char_start, char_end in spans:
            cropped_sentence, local_start, local_end = crop_around_target(
                original_sentence,
                char_start,
                char_end,
                window_chars=CONTEXT_WINDOW_CHARS,
            )

            inputs = tokenizer(
                cropped_sentence,
                return_tensors="pt",
                return_offsets_mapping=True,
                truncation=True,
                max_length=256,
            )

            offsets = inputs.pop("offset_mapping")[0].tolist()
            model_inputs = {key: value.to(device) for key, value in inputs.items()}

            output = model(**model_inputs, output_hidden_states=True)
            hidden_states = output.hidden_states[-1][0]

            token_positions: List[int] = []
            for tok_i, offset_pair in enumerate(offsets):
                start, end = int(offset_pair[0]), int(offset_pair[1])

                if start == end:
                    continue

                if start < local_end and end > local_start:
                    token_positions.append(tok_i)

            if token_positions:
                occurrence_vectors.append(
                    hidden_states[token_positions]
                    .mean(dim=0)
                    .detach()
                    .cpu()
                    .numpy()
                )
                audit["matched_span_occurrences"] += 1
            else:
                audit["span_fail_truncated_or_no_overlap"] += 1

        if occurrence_vectors:
            embeddings.append(np.mean(np.stack(occurrence_vectors), axis=0))
            kept_indices.append(sent_idx)
            audit["matched_span_vectors"] += 1
        elif allow_sentence_fallback:
            inputs = tokenizer(
                original_sentence,
                return_tensors="pt",
                truncation=True,
                max_length=256,
            ).to(device)
            output = model(**inputs, output_hidden_states=True)
            hidden_states = output.hidden_states[-1][0]

            embeddings.append(hidden_states.mean(dim=0).detach().cpu().numpy())
            kept_indices.append(sent_idx)
            audit["sentence_mean_fallback_vectors"] += 1

    if not embeddings:
        hidden_size = int(getattr(getattr(model, "config", None), "hidden_size", 768))
        return np.zeros((0, hidden_size), dtype=np.float32), [], audit

    return np.stack(embeddings).astype(np.float32), kept_indices, audit


def extraction_audit_fields(align_audit: Dict[str, int]) -> Dict[str, Any]:
    """Flatten extraction audit metrics for CSV output."""
    matched_sentence_vectors = int(align_audit.get("matched_span_vectors", 0))
    matched_occurrences = int(align_audit.get("matched_span_occurrences", 0))
    fallback = int(align_audit.get("sentence_mean_fallback_vectors", 0))
    candidate_occurrences = int(align_audit.get("candidate_occurrences", 0))
    candidate_sentences = int(align_audit.get("candidate_sentences", 0))
    span_fail = int(align_audit.get("span_fail_truncated_or_no_overlap", 0))

    span_match_rate = (
        matched_sentence_vectors / candidate_sentences
        if candidate_sentences > 0
        else float("nan")
    )

    return {
        "candidate_occurrences": candidate_occurrences,
        "candidate_sentences": candidate_sentences,
        "matched_span_occurrences": matched_occurrences,
        "matched_span_vectors": matched_sentence_vectors,
        "sentence_mean_fallback_vectors": fallback,
        "span_fail_truncated_or_no_overlap": span_fail,
        "span_match_rate": span_match_rate,
    }


def kmeans_labels(x_mat: np.ndarray, n_clusters: int, seed: int) -> np.ndarray:
    """Fit k-means with explicit reproducibility settings and return cluster labels."""
    kmeans_model = KMeans(
        n_clusters=n_clusters,
        random_state=seed,
        n_init=KMEANS_N_INIT,
        max_iter=KMEANS_MAX_ITER,
    )
    return kmeans_model.fit_predict(x_mat)

def best_k_by_indices(embeddings_mat: np.ndarray, random_seed: int) -> Tuple[int, pd.DataFrame]:
    """Select best k conservatively in [1, POLY_MAX_K], disallowing tiny clusters."""
    n_samples_local = embeddings_mat.shape[0]
    max_k_local = min(POLY_MAX_K, n_samples_local) if n_samples_local > 0 else 0
    rows_local: List[Dict[str, float]] = []

    if n_samples_local == 0:
        return 0, pd.DataFrame()

    for k_candidate in range(1, max_k_local + 1):
        if k_candidate == 1:
            rows_local.append(
                {
                    "k": 1.0,
                    "silhouette": float("nan"),
                    "calinski_harabasz": float("nan"),
                    "min_cluster_size": float(n_samples_local),
                }
            )
            continue

        try:
            cluster_labels = kmeans_labels(embeddings_mat, k_candidate, random_seed)
            _, cluster_counts = np.unique(cluster_labels, return_counts=True)
            min_cluster_size = int(cluster_counts.min()) if cluster_counts.size else 0

            if min_cluster_size < POLY_MIN_CLUSTER_SIZE:
                rows_local.append(
                    {
                        "k": float(k_candidate),
                        "silhouette": float("nan"),
                        "calinski_harabasz": float("nan"),
                        "min_cluster_size": float(min_cluster_size),
                    }
                )
                continue

            sil_value = (
                silhouette_score(embeddings_mat, cluster_labels)
                if len(set(cluster_labels)) > 1
                else -1.0
            )
            ch_value = calinski_harabasz_score(embeddings_mat, cluster_labels)

            rows_local.append(
                {
                    "k": float(k_candidate),
                    "silhouette": float(sil_value),
                    "calinski_harabasz": float(ch_value),
                    "min_cluster_size": float(min_cluster_size),
                }
            )
        except (ValueError, RuntimeError):
            rows_local.append(
                {
                    "k": float(k_candidate),
                    "silhouette": float("nan"),
                    "calinski_harabasz": float("nan"),
                    "min_cluster_size": float("nan"),
                }
            )

    result_df_local = pd.DataFrame(rows_local)

    if result_df_local.empty:
        return 1, pd.DataFrame()

    if result_df_local["k"].max() == 1:
        return 1, result_df_local

    valid_df_local = result_df_local.dropna(subset=["silhouette", "calinski_harabasz"])
    if valid_df_local.empty:
        return 1, result_df_local

    sil_idx_local = int(valid_df_local["silhouette"].idxmax())
    best_sil_value = float(valid_df_local.at[sil_idx_local, "silhouette"])
    tie_df_local = valid_df_local[valid_df_local["silhouette"] == best_sil_value]

    if len(tie_df_local) > 1:
        ch_idx_local = int(tie_df_local["calinski_harabasz"].idxmax())
        best_k_series_local = tie_df_local.loc[[ch_idx_local], "k"]
        best_k_value = float(best_k_series_local.to_numpy(dtype=float)[0])
    else:
        best_k_series_local = tie_df_local.loc[:, "k"]
        best_k_value = float(best_k_series_local.to_numpy(dtype=float)[0])

    return int(best_k_value), result_df_local

def bootstrap_ari_stability(
    x_mat: np.ndarray,
    n_clusters: int,
    seed: int,
    n_boot: int = 100,
) -> float:
    """Estimate clustering stability via bootstrap samples and the median adjusted Rand index."""
    if n_clusters <= 1 or x_mat.shape[0] < n_clusters:
        return float("nan")

    rng = np.random.default_rng(seed)
    base_labels = kmeans_labels(x_mat, n_clusters, seed)
    ari_values: List[float] = []

    for boot_idx in range(n_boot):
        indices = rng.integers(0, x_mat.shape[0], size=x_mat.shape[0])
        x_boot = x_mat[indices]
        ref_labels = base_labels[indices]

        try:
            boot_labels = kmeans_labels(x_boot, n_clusters, seed + boot_idx + 1)
            ari_values.append(float(adjusted_rand_score(ref_labels, boot_labels)))
        except (ValueError, RuntimeError):
            continue

    return float(np.nanmedian(ari_values)) if ari_values else float("nan")

def perm_test_separation(
    x_mat: np.ndarray,
    labels: np.ndarray,
    n_perm: int = 200,
    seed: int = 42,
) -> float:
    """Permutation test for separation: compare mean intra-cluster cosine vs shuffled labels."""
    if len(set(labels)) <= 1:
        return float("nan")
    rng = np.random.default_rng(seed)
    sims = squareform(1 - pdist(x_mat, metric="cosine"))

    cluster_means: List[float] = []
    for cluster_id in np.unique(labels):
        cluster_idx = np.where(labels == cluster_id)[0]
        if cluster_idx.size < 2:
            continue
        sub = np.take(np.take(sims, cluster_idx, axis=0), cluster_idx, axis=1)
        upper_mask = np.triu(np.ones(sub.shape, dtype=bool), 1)
        if not bool(np.any(upper_mask)):
            continue
        cluster_means.append(float(np.nanmean(sub[upper_mask])))
    obs = float(np.nanmean(cluster_means)) if cluster_means else float("nan")
    if np.isnan(obs):
        return float("nan")

    more = 0
    for _ in range(n_perm):
        shuffled = rng.permutation(labels)
        cluster_means_perm: List[float] = []
        for cluster_id in np.unique(shuffled):
            cluster_idx_perm = np.where(shuffled == cluster_id)[0]
            if cluster_idx_perm.size < 2:
                continue
            sub = np.take(np.take(sims, cluster_idx_perm, axis=0), cluster_idx_perm, axis=1)
            upper_mask = np.triu(np.ones(sub.shape, dtype=bool), 1)
            if not bool(np.any(upper_mask)):
                continue
            cluster_means_perm.append(float(np.nanmean(sub[upper_mask])))
        val = float(np.nanmean(cluster_means_perm)) if cluster_means_perm else float("nan")
        if not np.isnan(val) and val >= obs:
            more += 1
    p_value = (more + 1) / (n_perm + 1)
    return float(p_value)

def representatives_and_centroids(
    x_mat: np.ndarray,
    labels: np.ndarray,
    sentences: List[str],
    idxs: List[int],
    topn: int = REP_SENTENCES_PER_CLUSTER,
) -> Tuple[pd.DataFrame, Dict[int, np.ndarray], Dict[int, int]]:
    """Compute cluster centroids and select representative sentences per cluster."""
    representatives: List[Dict[str, Any]] = []
    centroids: Dict[int, np.ndarray] = {}
    sizes: Dict[int, int] = {}

    for cluster_id in sorted(set(labels)):
        cluster_indices = np.where(labels == cluster_id)[0]
        sizes[cluster_id] = int(cluster_indices.size)
        sub = x_mat[cluster_indices]
        centroid = sub.mean(axis=0)
        centroids[cluster_id] = centroid

        distances = [1.0 - cosine(vec, centroid) for vec in sub]

        order = np.argsort(distances)
        order_indices: List[int] = [int(i) for i in order]

        max_rep = min(len(order_indices), topn)

        for rank, idx_in_cluster in enumerate(order_indices[:max_rep], start=1):
            global_sent_idx = int(idxs[cluster_indices[idx_in_cluster]])
            sent = sentences[global_sent_idx]

            representatives.append(
                {
                    "cluster": int(cluster_id),
                    "rank": int(rank),
                    "distance": float(distances[idx_in_cluster]),
                    "sentence": sent,
                }
            )

    rep_df = pd.DataFrame(representatives)
    return rep_df, centroids, sizes


def _safe_model_key(model_name: str) -> str:
    """Return a filesystem-safe model key."""
    return re.sub(r"[^A-Za-z0-9_.-]+", "_", model_name)

def polysemy_prov_dir(term: str, mode: str, model_name: str) -> Path:
    """Return the provenance directory for one term, one corpus mode, and one model."""
    return PROV / f"{term}_{mode.upper()}_{_safe_model_key(model_name)}"

def load_requested_bert_models(mode: str) -> Dict[str, Tuple[Any, BertTokenizer]]:
    """Load the explicitly requested BERT models for one mode.

    No hidden substitution is allowed.
    """
    out: Dict[str, Tuple[BertForMaskedLM, BertTokenizer]] = {}

    if USE_BERT_BASE:
        model_base, tok_base, _ = load_bert_base()
        out["bert_base"] = (model_base, tok_base)

    if USE_BERT_FINETUNED:
        loaded_ft = load_bert_finetuned(mode)
        if not loaded_ft:
            raise RuntimeError(
                f"Fine-tuned BERT was requested for mode={mode}, but no valid checkpoint was resolved."
            )
        model_ft, tok_ft, _, _ = loaded_ft
        out["bert_ft"] = (model_ft, tok_ft)

    if not out:
        raise RuntimeError(f"No BERT models are enabled for mode={mode}.")

    return out

def write_resolved_bert_checkpoint_audit() -> None:
    """Write the resolved FULL/CLEAN fine-tuned checkpoint paths and loaded-state hashes."""
    rows: List[Dict[str, Any]] = []

    for mode in ("FULL", "CLEAN"):
        loaded_ft = load_bert_finetuned(mode)
        if not loaded_ft:
            continue

        _, _, best_epoch, ckpt = loaded_ft

        if (ckpt / "adapter_model.safetensors").is_file():
            load_kind = "adapter"
            state_path = ckpt / "adapter_model.safetensors"
        else:
            load_kind = "merged_model"
            state_path = ckpt / "model.safetensors"
            if not state_path.is_file():
                state_path = ckpt / "pytorch_model.bin"

        rows.append(
            {
                "mode": mode,
                "load_kind": load_kind,
                "checkpoint_dir": str(ckpt.resolve()),
                "best_epoch": int(best_epoch),
                "config_sha": sha256_file(ckpt / "config.json"),
                "state_file": state_path.name if state_path.exists() else "NA",
                "state_sha": sha256_file(state_path),
                "tokenizer_json_sha": sha256_file(ckpt / "tokenizer.json"),
                "vocab_sha": sha256_file(ckpt / "vocab.txt"),
            }
        )

    if not rows:
        LOGGER.warning("No resolved fine-tuned BERT checkpoints were available for audit.")
        return

    audit_df = pd.DataFrame(rows)
    audit_df.to_csv(RES / "bert_checkpoint_audit.csv", index=False)

    if len(audit_df) == 2 and audit_df["state_sha"].nunique() == 1:
        LOGGER.warning(
            "FULL and CLEAN resolved to the same loaded BERT state. "
            "Check your model caches before interpreting FULL/CLEAN classification outputs."
        )

def save_polysemy_provenance(
    term: str,
    mode: str,
    model_name: str,
    x_mat: np.ndarray,
    labels: np.ndarray,
    centroids: Dict[int, np.ndarray],
    sizes: Dict[int, int],
) -> None:
    """Persist embeddings, labels, centroids, and cluster sizes for a term, mode, and model."""
    base_dir = polysemy_prov_dir(term, mode, model_name)
    base_dir.mkdir(exist_ok=True, parents=True)
    np.save(str(base_dir / "embeddings.npy"), x_mat)
    np.save(str(base_dir / "labels.npy"), labels)
    for cluster_id, centroid in centroids.items():
        np.save(str(base_dir / f"centroid_c{cluster_id}.npy"), centroid)
    sizes_payload = {int(key): int(value) for key, value in sizes.items()}
    (base_dir / "sizes.json").write_text(
        json.dumps(sizes_payload, ensure_ascii=False),
        encoding="utf-8",
    )

def load_polysemy_embeddings(
    term: str,
    mode: str = "FULL",
    model_name: str = "bert_ft",
) -> Optional[np.ndarray]:
    """Load stored contextual embeddings for a term, mode, and model, if present."""
    base_dir = polysemy_prov_dir(term, mode, model_name)
    path = base_dir / "embeddings.npy"
    if not path.is_file():
        return None
    arr = np.load(str(path))
    if arr.ndim != 2 or arr.shape[0] == 0:
        return None
    return arr.astype(np.float32)

def plot_tsne(x_mat: np.ndarray, labels: np.ndarray, title: str, outfile: Path) -> None:
    """Project embeddings with t-SNE and save a scatterplot coloured by cluster labels."""
    if x_mat.shape[0] < 2:
        return
    perplexity = max(5, min(30, x_mat.shape[0] - 1))
    reduced = TSNE(
        n_components=2,
        random_state=SEED,
        perplexity=perplexity,
    ).fit_transform(x_mat)
    plt.figure(figsize=(8, 7))
    sns.scatterplot(
        x=reduced[:, 0],
        y=reduced[:, 1],
        hue=labels,
        palette="viridis",
        s=60,
        legend="full",
    )
    plt.title(title)
    plt.tight_layout()
    _save_png_svg(outfile, dpi=160)
    plt.close()

def run_polysemy_for_mode(mode: str, term_list: List[str]) -> Dict[str, Dict[str, Any]]:
    """Run polysemy pipeline for all requested BERT models in a given mode."""
    out_summary: Dict[str, Dict[str, Any]] = {}

    corpus = resolve_bert_path(mode)
    if not corpus.exists():
        LOGGER.error("BERT corpus not found for %s: %s", mode, corpus)
        return out_summary

    sentences = read_sentences(corpus)

    try:
        bert_models = load_requested_bert_models(mode)
    except RuntimeError as exc:
        LOGGER.error("%s", exc)
        return out_summary

    for model_name, model_pack in bert_models.items():
        model, tokenizer = model_pack
        rows_for_table: List[Dict[str, Any]] = []

        for term in term_list:
            banner(f"CONTEXTUAL CLUSTER RECOVERY {mode} — {term} — {model_name}")
            x_mat, idxs, align_audit = contextual_embeddings_for(
                term,
                sentences,
                model,
                tokenizer,
                allow_sentence_fallback=False,
            )

            if x_mat.shape[0] < POLY_MIN_INST:
                LOGGER.info(
                    " -> %s [%s]: not enough matched contexts (%d) — recording as insufficient and skipping clustering",
                    term,
                    model_name,
                    x_mat.shape[0],
                )

                audit_fields = extraction_audit_fields(align_audit)

                rows_for_table.append(
                    {
                        "term": term,
                        "mode": mode,
                        "model": model_name,
                        "best_k": 0,
                        "contexts": int(x_mat.shape[0]),
                        "candidate_occurrences": audit_fields["candidate_occurrences"],
                        "candidate_sentences": audit_fields["candidate_sentences"],
                        "matched_span_occurrences": audit_fields["matched_span_occurrences"],
                        "matched_span_vectors": audit_fields["matched_span_vectors"],
                        "sentence_mean_fallback_vectors": audit_fields["sentence_mean_fallback_vectors"],
                        "span_fail_truncated_or_no_overlap": audit_fields["span_fail_truncated_or_no_overlap"],
                        "span_match_rate": audit_fields["span_match_rate"],
                        "stability_ari_median": float("nan"),
                        "perm_p_value": float("nan"),
                        "tsne_png": "",
                        "reps_csv": "",
                        "scores_csv": "",
                    }
                )

                out_summary[f"{model_name}::{term}"] = {
                    "model": model_name,
                    "term": term,
                    "k": 0,
                    "centroids": {},
                    "sizes": {},
                }
                continue

            k_best, score_table = best_k_by_indices(x_mat, SEED)
            stability = float("nan")
            p_value = float("nan")
            labels = np.zeros(x_mat.shape[0], dtype=int)

            if k_best >= 1:
                labels = (
                    kmeans_labels(x_mat, max(k_best, 1), SEED)
                    if k_best > 1
                    else np.zeros(x_mat.shape[0], dtype=int)
                )
                if RUN_STATS and k_best > 1:
                    stability = bootstrap_ari_stability(x_mat, k_best, SEED, n_boot=100)
                    p_value = perm_test_separation(x_mat, labels, n_perm=300, seed=SEED)

            reps_df, centroids, sizes = representatives_and_centroids(
                x_mat,
                labels,
                sentences,
                idxs,
                REP_SENTENCES_PER_CLUSTER,
            )

            score_path = RES / f"polysemy_scores_{term}_{model_name}_{mode.lower()}.csv"
            if not score_table.empty:
                score_table.to_csv(str(score_path), index=False)

            rep_path = RES / f"polysemy_representatives_{term}_{model_name}_{mode.lower()}.csv"
            reps_df.to_csv(str(rep_path), index=False)

            tsne_path = FIG / f"polysemy_tsne_{term}_{model_name}_{mode.lower()}_k{k_best}.png"
            tsne_written = False
            if not (SUPPRESS_TSNE_FOR_K1 and int(k_best) == 1):

                    f"{term} — {mode} — k={k_best} ({model_name})",

                    if k_val < 1 or k_val == k_best:
                        continue
                    if SUPPRESS_TSNE_FOR_K1 and int(k_val) == 1:
                        continue
                    labs = (
                        kmeans_labels(x_mat, k_val, SEED)
                        if k_val > 1
                        else np.zeros(x_mat.shape[0], dtype=int)
                    )
                    path = FIG / f"polysemy_tsne_{term}_{model_name}_{mode.lower()}_k{k_val}.png"
                    plot_tsne(
                        x_mat,
                        labs,
                        f"{term} — {mode} — k={k_val} ({model_name})",
                        path,
                    )

            save_polysemy_provenance(term, mode, model_name, x_mat, labels, centroids, sizes)

            audit_fields = extraction_audit_fields(align_audit)

            rows_for_table.append(
                {
                    "term": term,
                    "mode": mode,
                    "model": model_name,
                    "best_k": int(k_best),
                    "contexts": int(x_mat.shape[0]),
                    "candidate_occurrences": audit_fields["candidate_occurrences"],
                    "candidate_sentences": audit_fields["candidate_sentences"],

        if rows_for_table:
            summary_path = RES / f"polysemy_summary_{model_name}_{mode.lower()}.csv"
            pd.DataFrame(rows_for_table).to_csv(str(summary_path), index=False)

    return out_summary

def per_model_vectors(
    targets: List[str],
    labels: List[str],
    mode: str = "FULL",
) -> Dict[str, Dict[str, Dict[str, np.ndarray]]]:
    """
    Build per-model vectors for a given corpus mode ("FULL" or "CLEAN").

    No hidden omission of requested fine-tuned BERT is allowed.
    """
    out: Dict[str, Dict[str, Dict[str, np.ndarray]]] = {}
    mode_tag = mode.upper()

    if USE_W2V_CBOW:
        w2v_cbow_model = load_w2v(mode_tag, "cbow")
        if w2v_cbow_model is not None:
            target_vecs = {t: w2v_vec(w2v_cbow_model, t) for t in targets}
            label_vecs = {lbl: w2v_vec(w2v_cbow_model, lbl) for lbl in labels}
            out["w2v_cbow"] = {"targets": target_vecs, "labels": label_vecs}

    if USE_W2V_SG:
        w2v_sg_model = load_w2v(mode_tag, "sg")
        if w2v_sg_model is not None:
            target_vecs = {t: w2v_vec(w2v_sg_model, t) for t in targets}
            label_vecs = {lbl: w2v_vec(w2v_sg_model, lbl) for lbl in labels}
            out["w2v_sg"] = {"targets": target_vecs, "labels": label_vecs}

    if USE_BERT_BASE:
        bert_model, bert_tokenizer, _ = load_bert_base()
        target_vecs = {
            t: bert_vec(bert_model, bert_tokenizer, t, is_sentence=False) for t in targets
        }
        label_vecs = {
            lbl: bert_vec(bert_model, bert_tokenizer, lbl, is_sentence=True)
            for lbl in labels
        }
        out["bert_base"] = {"targets": target_vecs, "labels": label_vecs}

    if USE_BERT_FINETUNED:
        loaded_ft = load_bert_finetuned(mode_tag)
        if not loaded_ft:
            raise RuntimeError(
                f"Fine-tuned BERT was requested for zero-shot in mode={mode_tag}, but no valid checkpoint was resolved."
            )
        model_ft, tok_ft, _, _ = loaded_ft
        target_vecs = {
            t: bert_vec(model_ft, tok_ft, t, is_sentence=False) for t in targets
        }
        label_vecs = {
            lbl: bert_vec(model_ft, tok_ft, lbl, is_sentence=True)
            for lbl in labels
        }
        out["bert_ft"] = {"targets": target_vecs, "labels": label_vecs}

    return out

def similarity_table(
    model_block: Dict[str, Dict[str, np.ndarray]],
    targets: List[str],
    labels: List[str],
) -> pd.DataFrame:
    """Build a term×label similarity matrix for a single model block."""
    rows: List[Dict[str, Any]] = []
    for term in targets:
        term_vec = model_block["targets"].get(term)
        for label in labels:
            label_vec = model_block["labels"].get(label)
            if (
                term_vec is None
                or label_vec is None
                or (isinstance(term_vec, float) and np.isnan(term_vec))
            ):
                sim_val = float("nan")
            else:
                sim_val = cosine(term_vec, label_vec)  # type: ignore[arg-type]
            rows.append({"term": term, "label": label, "similarity": sim_val})
    df = pd.DataFrame(rows)
    return (
        df.pivot_table(
            index="term",
            columns="label",
            values="similarity",
            aggfunc="mean",
        )
        .sort_index()
    )


def to_flags(scores: pd.DataFrame) -> pd.DataFrame:
    """Threshold similarity scores into a binary term×label table using configured rules."""
    selected_percentile: set[Tuple[Hashable, Hashable]] = set()

    if ML_USE_ROWWISE_PERCENTILE:
        threshs = scores.apply(
            lambda row_values: float(
                np.nanpercentile(
                    row_values.values.astype(float),
                    ML_ROWWISE_PERCENTILE,
                )
            )
            if np.isfinite(row_values.values.astype(float)).any()
            else float("nan"),
            axis=1,
        )

        for term, row_series in scores.iterrows():
            thr_raw = threshs.get(term)
            if isinstance(thr_raw, (float, int, np.floating)):
                thr_val = float(thr_raw)
            else:
                thr_val = float("nan")
            for label, score_val in row_series.items():
                score_float = float(score_val) if pd.notna(score_val) else float("nan")
                if not np.isnan(score_float) and score_float >= max(ML_MIN_SIM, thr_val):
                    selected_percentile.add((term, label))

    selected_topk: set[Tuple[Hashable, Hashable]] = set()
    if (not ML_USE_ROWWISE_PERCENTILE) and ML_TOPK > 0:
        for term, row_series in scores.iterrows():
            values = row_series.dropna().astype(float).sort_values(ascending=False)
            keep_labels = values.head(ML_TOPK).index.tolist()
            for label in keep_labels:
                if values.loc[label] >= ML_MIN_SIM:
                    selected_topk.add((term, label))

    chosen = selected_percentile | selected_topk
    flags = pd.DataFrame(0, index=scores.index, columns=scores.columns, dtype=int)
    for term, label in chosen:
        if term in flags.index and label in flags.columns:
            flags.loc[term, label] = 1
    return flags


def consensus(scores_by_model: Dict[str, pd.DataFrame]) -> pd.DataFrame:
    """Aggregate per-model similarity tables into a single consensus similarity table."""
    label_index = None
    term_index = None
    for df in scores_by_model.values():
        if df is not None and not df.empty:
            label_index = df.columns
            term_index = df.index
            break
    if label_index is None or term_index is None:
        return pd.DataFrame()

    arrays: List[np.ndarray] = []
    for df in scores_by_model.values():
        if df is None or df.empty:
            continue
        arr = df.reindex(index=term_index, columns=label_index).to_numpy(dtype=float)
        arrays.append(arr)
    if not arrays:
        return pd.DataFrame(index=term_index, columns=label_index)

    stacked = np.stack(arrays, axis=2)  # [terms, labels, models]
    valid_counts = np.isfinite(stacked).sum(axis=2)
    if CONS_STRATEGY == "median":
        aggregated = np.nanmedian(stacked, axis=2)
    else:
        aggregated = np.nanmean(stacked, axis=2)
    aggregated[valid_counts < REQUIRE_MIN_MODELS] = np.nan
    return pd.DataFrame(aggregated, index=term_index, columns=label_index)


def row_zscore(row: pd.Series) -> pd.Series:
    """Row-wise z-score with NaN-safe mean/std; returns centred values if variance is zero."""
    vals = row.values.astype(float)
    m = float(np.nanmean(vals))
    s = float(np.nanstd(vals, ddof=1))
    if not np.isfinite(s) or s == 0.0:
        return row - m
    return (row - m) / s


def rowwise_percent_scale(df: pd.DataFrame) -> pd.DataFrame:
    """
    For each row, rescale finite values linearly to [0, 100].

    0% = weakest label for that term, 100% = strongest. Degenerate rows (all equal / all NaN)
    become 100 for finite entries.
    """

    def _scale_row(r: pd.Series) -> pd.Series:
        vals = r.values.astype(float)
        finite = np.isfinite(vals)
        out = np.full_like(vals, np.nan, dtype=float)
        if not finite.any():
            return pd.Series(out, index=r.index)

        v_min = float(np.nanmin(vals[finite]))
        v_max = float(np.nanmax(vals[finite]))

        if not np.isfinite(v_max) or v_max <= v_min:
            out[finite] = 100.0
        else:
            out[finite] = (vals[finite] - v_min) * 100.0 / (v_max - v_min)

        return pd.Series(out, index=r.index)

    # Build rows explicitly so the return type is unambiguously a DataFrame
    scaled_rows: list[pd.Series] = [_scale_row(row) for _, row in df.iterrows()]
    out_df = pd.DataFrame(scaled_rows, index=df.index, columns=df.columns)
    return out_df


def jaccard_rows(df_a: pd.DataFrame, df_b: pd.DataFrame) -> pd.DataFrame:
    """Row-wise Jaccard between two binary tables with same shape."""
    if df_a.empty or df_b.empty:
        return pd.DataFrame()
    df_a_int = df_a.fillna(0).astype(int)
    df_b_int = df_b.fillna(0).astype(int)
    rows: List[Dict[str, Any]] = []
    for term in df_a_int.index:
        labels_a = set(df_a_int.columns[df_a_int.loc[term] == 1])
        labels_b = set(df_b_int.columns[df_b_int.loc[term] == 1])
        inter = len(labels_a & labels_b)
        union = len(labels_a | labels_b) if (labels_a or labels_b) else 1
        rows.append({"term": term, "jaccard": inter / union})
    return pd.DataFrame(rows)


def compare_flag_tables(
    df_a: pd.DataFrame,
    df_b: pd.DataFrame,
    name_a: str,
    name_b: str,
) -> None:
    """Log simple agreement statistics between two binary flag tables."""
    if df_a.empty or df_b.empty:
        LOGGER.warning("compare_flag_tables: one or both tables are empty (%s, %s)", name_a, name_b)
        return

    common_terms = sorted(set(df_a.index) & set(df_b.index))
    common_labels = sorted(set(df_a.columns) & set(df_b.columns))
    if not common_terms or not common_labels:
        LOGGER.warning("compare_flag_tables: no common index/columns between %s and %s", name_a, name_b)
        return

    a = (
        df_a.reindex(index=common_terms, columns=common_labels)
        .fillna(0)
        .astype(int)
        .to_numpy()
    )
    b = (
        df_b.reindex(index=common_terms, columns=common_labels)
        .fillna(0)
        .astype(int)
        .to_numpy()
    )

    eq_mask = np.equal(a, b)
    agree = int(np.count_nonzero(eq_mask))
    total = int(a.size)

    both_one_mask = (a == 1) & (b == 1)
    either_one_mask = (a == 1) | (b == 1)

    both_count = int(np.count_nonzero(both_one_mask))
    either_count = int(np.count_nonzero(either_one_mask))
    if either_count <= 0:
        either_count = 1
    jaccard = float(both_count / either_count)

    LOGGER.info(
        "Flag agreement %s vs %s: raw=%d/%d (%.3f), Jaccard=%.3f over term×label cells",
        name_a,
        name_b,
        agree,
        total,
        agree / total,
        jaccard,
    )

# SECTION 9: ZERO-SHOT BOOTSTRAP FROM CONTEXTS
# =========================
def bootstrap_zero_shot_from_contexts(
    mode: str = "FULL",
    n_boot: int = ZS_BOOT_ITERS,
    model_name: Optional[str] = None,
) -> Tuple[pd.DataFrame, pd.DataFrame]:
    """
    Bootstrap term×label similarities using contextual embeddings.

    Returns
    -------
    summary_df : rows (term, label) with observed sim, CI, etc.
    pair_df    : rows (term, top_label, other_label) with p-values + FDR.
    """
    try:
        bert_models = load_requested_bert_models(mode)
    except RuntimeError as exc:
        LOGGER.error("%s", exc)
        return pd.DataFrame(), pd.DataFrame()

    chosen_model = model_name or preferred_bert_model_name(bert_models)
    if chosen_model not in bert_models:
        LOGGER.error("ZS bootstrap requested unknown model '%s' for mode=%s", chosen_model, mode)
        return pd.DataFrame(), pd.DataFrame()

    model, tokenizer = bert_models[chosen_model]
    LOGGER.info("ZS bootstrap using %s for labels (%s)", chosen_model, mode)

    label_vecs: Dict[str, Optional[np.ndarray]] = {
        lbl: bert_vec(model, tokenizer, lbl, is_sentence=True) for lbl in CATEGORY_LABELS
    }

    rng = np.random.default_rng(SEED)
    summary_rows: List[Dict[str, Any]] = []
    pair_rows: List[Dict[str, Any]] = []

    for term in TARGETS:
        x_mat = load_polysemy_embeddings(term, mode, model_name=chosen_model)
        if x_mat is None or x_mat.shape[0] < POLY_MIN_INST:
            LOGGER.info(
                "ZS bootstrap: %s (%s, %s) has too few contexts (%s) — skipping",
                term,
                mode,
                chosen_model,
                0 if x_mat is None else x_mat.shape[0],
            )
            continue

        n_ctx = x_mat.shape[0]
        term_vec_obs = x_mat.mean(axis=0)

        obs_sims: Dict[str, float] = {}
        for label, lbl_vec in label_vecs.items():
            if lbl_vec is None or (isinstance(lbl_vec, float) and np.isnan(lbl_vec)):
                obs_sims[label] = float("nan")
            else:
                obs_sims[label] = float(cosine(term_vec_obs, lbl_vec))

        sims_boot: Dict[str, List[float]] = {lbl: [] for lbl in CATEGORY_LABELS}
        for _ in range(n_boot):
            sample_idx = rng.integers(0, n_ctx, size=n_ctx)
            term_vec_b = x_mat[sample_idx].mean(axis=0)
            for label, lbl_vec in label_vecs.items():
                if lbl_vec is None or (isinstance(lbl_vec, float) and np.isnan(lbl_vec)):
                    continue
                sims_boot[label].append(float(cosine(term_vec_b, lbl_vec)))

        for label in CATEGORY_LABELS:
            vals = np.asarray(sims_boot[label], dtype=float)
            if vals.size == 0:
                mean_b = med_b = sd_b = q_lo = q_hi = float("nan")
            else:
                mean_b = float(vals.mean())
                med_b = float(np.median(vals))
                sd_b = float(vals.std(ddof=1)) if vals.size > 1 else 0.0
                q_lo, q_hi = np.percentile(vals, [2.5, 97.5]).tolist()
            summary_rows.append(
                {
                    "term": term,
                    "mode": mode,
                    "model": chosen_model,
                    "label": label,
                    "contexts": int(n_ctx),
                    "sim_observed": obs_sims.get(label, float("nan")),
                    "sim_boot_mean": mean_b,
                    "sim_boot_median": med_b,
                    "sim_boot_sd": sd_b,
                    "sim_ci_95_low": q_lo,
                    "sim_ci_95_high": q_hi,
                }
            )

        valid_items = [(lbl, s) for lbl, s in obs_sims.items() if np.isfinite(s)]
        if not valid_items:
            continue
        top_label, top_val = max(valid_items, key=lambda x: x[1])
        vals_top = np.asarray(sims_boot[top_label], dtype=float)

        for other_label, other_val in valid_items:
            if other_label == top_label:
                continue
            vals_other = np.asarray(sims_boot[other_label], dtype=float)
            if vals_top.size == 0 or vals_other.size == 0:
                p2 = float("nan")
                diff_obs = float(top_val - other_val)
            else:
                k = min(vals_top.size, vals_other.size)
                diffs = vals_top[:k] - vals_other[:k]
                diff_obs = float(top_val - other_val)
                p2 = (int(np.sum(np.abs(diffs) >= abs(diff_obs))) + 1) / (k + 1)
            pair_rows.append(
                {
                    "term": term,
                    "mode": mode,
                    "model": chosen_model,
                    "top_label": top_label,
                    "other_label": other_label,
                    "diff_observed": diff_obs,
                    "p_boot_2sided": p2,
                    "contexts": int(n_ctx),
                }
            )

    summary_df = pd.DataFrame(summary_rows)
    pair_df = pd.DataFrame(pair_rows)

    if not pair_df.empty and "p_boot_2sided" in pair_df.columns:
        p_series = pair_df["p_boot_2sided"]
        valid_mask = p_series.notna()
        valid_p = p_series[valid_mask].astype(float)
        pvals = valid_p.tolist()
        if pvals:
            adj_bh = fdr_bh(pvals)
            adj_holm = holm_bonferroni(pvals)
            pair_df.loc[valid_p.index, "p_fdr_bh"] = adj_bh
            pair_df.loc[valid_p.index, "p_holm"] = adj_holm
            pair_df["p_fdr_bh"] = pair_df["p_fdr_bh"].astype(float)
            pair_df["p_holm"] = pair_df["p_holm"].astype(float)
            pair_df["sig_fdr_bh_0.05"] = pair_df["p_fdr_bh"] < 0.05
            pair_df["sig_holm_0.05"] = pair_df["p_holm"] < 0.05

    return summary_df, pair_df


def bootstrap_concept_alignment_from_contexts(
    mode: str = "FULL",
    n_boot: int = ZS_BOOT_ITERS,
    model_name: Optional[str] = None,
) -> Tuple[pd.DataFrame, pd.DataFrame]:
    """
    Bootstrap term×concept similarities using contextual embeddings.

    This bootstrap must use the same bounded Paper 4 concept surface as the main
    engawa module, rather than re-reading a broader global config block.
    """
    try:
        bert_models = load_requested_bert_models(mode)
    except RuntimeError as exc:
        LOGGER.error("%s", exc)
        return pd.DataFrame(), pd.DataFrame()

    chosen_model = model_name or preferred_bert_model_name(bert_models)
    if chosen_model not in bert_models:
        LOGGER.error(
            "Concept bootstrap requested unknown model '%s' for mode=%s",
            chosen_model,
            mode,
        )
        return pd.DataFrame(), pd.DataFrame()

    model, tokenizer = bert_models[chosen_model]
    LOGGER.info("Concept bootstrap using %s for concepts (%s)", chosen_model, mode)

    # IMPORTANT:
    # Use the already-prepared, Paper 4-scoped concept surface, not a fresh
    # direct read from cfg["concept_definitions"].
    concept_defs: Dict[str, List[str]] = CONCEPT_DEFINITIONS
    if not concept_defs:
        LOGGER.warning(
            "CONCEPT_DEFINITIONS is empty after Paper 4 config resolution; "
            "skipping concept bootstrap."
        )
        return pd.DataFrame(), pd.DataFrame()

    concept_vecs: Dict[str, Optional[np.ndarray]] = {}
    for concept_name, anchors in concept_defs.items():
        anchor_vecs: List[np.ndarray] = []
        for anchor in anchors:
            v = bert_vec(model, tokenizer, str(anchor), is_sentence=False)
            if v is None:
                continue
            if isinstance(v, float) and math.isnan(v):
                continue
            anchor_vecs.append(v)

        if anchor_vecs:
            concept_vecs[concept_name] = np.mean(np.stack(anchor_vecs, axis=0), axis=0)
        else:
            LOGGER.warning(
                "No valid vectors for concept '%s' (anchors=%s) — concept treated as missing",
                concept_name,
                anchors,
            )
            concept_vecs[concept_name] = None

    concept_names: List[str] = list(concept_vecs.keys())
    if not concept_names:
        LOGGER.warning("No concept names available after Paper 4 concept-vector construction.")
        return pd.DataFrame(), pd.DataFrame()

    rng = np.random.default_rng(SEED)
    summary_rows: List[Dict[str, Any]] = []
    pair_rows: List[Dict[str, Any]] = []

    for term in CONCEPT_ALIGNMENT_TARGETS:
        x_mat = load_polysemy_embeddings(term, mode, model_name=chosen_model)
        if x_mat is None or x_mat.shape[0] < POLY_MIN_INST:
            LOGGER.info(
                "Concept bootstrap: %s (%s, %s) has too few contexts (%s) — skipping",
                term,
                mode,
                chosen_model,
                0 if x_mat is None else x_mat.shape[0],
            )
            continue

        n_ctx = x_mat.shape[0]
        term_vec_obs = x_mat.mean(axis=0)

        obs_sims: Dict[str, float] = {}
        for concept_name, c_vec in concept_vecs.items():
            if c_vec is None or (isinstance(c_vec, float) and math.isnan(c_vec)):
                obs_sims[concept_name] = float("nan")
            else:
                obs_sims[concept_name] = float(cosine(term_vec_obs, c_vec))

        sims_boot: Dict[str, List[float]] = {c: [] for c in concept_names}
        for _ in range(n_boot):
            sample_idx = rng.integers(0, n_ctx, size=n_ctx)
            term_vec_b = x_mat[sample_idx].mean(axis=0)
            for concept_name, c_vec in concept_vecs.items():
                if c_vec is None or (isinstance(c_vec, float) and math.isnan(c_vec)):
                    continue
                sims_boot[concept_name].append(float(cosine(term_vec_b, c_vec)))

        for concept_name in concept_names:
            vals = np.asarray(sims_boot[concept_name], dtype=float)
            if vals.size == 0:
                mean_b = med_b = sd_b = q_lo = q_hi = float("nan")
            else:
                mean_b = float(vals.mean())
                med_b = float(np.median(vals))
                sd_b = float(vals.std(ddof=1)) if vals.size > 1 else 0.0
                q_lo, q_hi = np.percentile(vals, [2.5, 97.5]).tolist()

            summary_rows.append(
                {
                    "term": term,
                    "mode": mode,
                    "model": chosen_model,
                    "concept": concept_name,
                    "contexts": int(n_ctx),
                    "sim_observed": obs_sims.get(concept_name, float("nan")),
                    "sim_boot_mean": mean_b,
                    "sim_boot_median": med_b,
                    "sim_boot_sd": sd_b,
                    "sim_ci_95_low": q_lo,
                    "sim_ci_95_high": q_hi,
                }
            )

        valid_items = [(c, s) for c, s in obs_sims.items() if np.isfinite(s)]
        if not valid_items:
            continue

        top_concept, top_val = max(valid_items, key=lambda x: x[1])
        vals_top = np.asarray(sims_boot[top_concept], dtype=float)

        for other_concept, other_val in valid_items:
            if other_concept == top_concept:
                continue

            vals_other = np.asarray(sims_boot[other_concept], dtype=float)
            if vals_top.size == 0 or vals_other.size == 0:
                p2 = float("nan")
                diff_obs = float(top_val - other_val)
            else:
                k = min(vals_top.size, vals_other.size)
                diffs = vals_top[:k] - vals_other[:k]
                diff_obs = float(top_val - other_val)
                p2 = (int(np.sum(np.abs(diffs) >= abs(diff_obs))) + 1) / (k + 1)

            pair_rows.append(
                {
                    "term": term,
                    "mode": mode,
                    "model": chosen_model,
                    "top_concept": top_concept,
                    "other_concept": other_concept,
                    "diff_observed": diff_obs,
                    "p_boot_2sided": p2,
                    "contexts": int(n_ctx),
                }
            )

    summary_df = pd.DataFrame(summary_rows)
    pair_df = pd.DataFrame(pair_rows)

    if not pair_df.empty and "p_boot_2sided" in pair_df.columns:
        p_series = pair_df["p_boot_2sided"]
        valid_mask = p_series.notna()
        valid_p = p_series[valid_mask].astype(float)
        pvals = valid_p.tolist()
        if pvals:
            adj_bh = fdr_bh(pvals)
            adj_holm = holm_bonferroni(pvals)
            pair_df.loc[valid_p.index, "p_fdr_bh"] = adj_bh
            pair_df.loc[valid_p.index, "p_holm"] = adj_holm
            pair_df["p_fdr_bh"] = pair_df["p_fdr_bh"].astype(float)
            pair_df["p_holm"] = pair_df["p_holm"].astype(float)
            pair_df["sig_fdr_bh_0.05"] = pair_df["p_fdr_bh"] < 0.05
            pair_df["sig_holm_0.05"] = pair_df["p_holm"] < 0.05

    return summary_df, pair_df

def sense_centroids_load(
    term: str,
    mode: str = "FULL",
    model_name: str = "bert_ft",
) -> Optional[Tuple[Dict[int, np.ndarray], Dict[int, int]]]:
    """Load stored sense centroids and cluster sizes for a term, mode, and model, if present."""
    base_dir = polysemy_prov_dir(term, mode, model_name)
    if not base_dir.exists():
        return None
    centroids: Dict[int, np.ndarray] = {}
    sizes: Dict[int, int] = {}
    for path in base_dir.glob("centroid_c*.npy"):
        cluster_id = int(path.stem.split("c")[-1])
        centroids[cluster_id] = np.load(str(path))
    sizes_path = base_dir / "sizes.json"
    if sizes_path.exists():
        sizes = {
            int(key): int(value)
            for key, value in json.loads(
                sizes_path.read_text(encoding="utf-8")
            ).items()
        }
    return centroids, sizes

def sense_aware_scores(
    centroids: Dict[int, np.ndarray],
    sizes: Dict[int, int],
    label_vecs: Dict[str, Optional[np.ndarray]],
    strategy: str = "weighted",
) -> Dict[str, float]:
    """Compute per-label sense-aware similarity from centroids and label vectors."""
    scores_by_label: Dict[str, float] = {}
    if not centroids:
        return {label: float("nan") for label in label_vecs.keys()}
    total = sum(sizes.get(cluster_id, 1) for cluster_id in centroids)
    if total <= 0:
        total = float(len(centroids))
    weights = {
        cluster_id: sizes.get(cluster_id, 1) / total for cluster_id in centroids
    }
    for label, label_vec in label_vecs.items():
        if label_vec is None or (isinstance(label_vec, float) and np.isnan(label_vec)):
            scores_by_label[label] = float("nan")
            continue
        sims: List[float] = [
            cosine(centroid_vec, label_vec) for centroid_vec in centroids.values()
        ]
        if not sims:
            scores_by_label[label] = float("nan")
            continue
        if strategy == "weighted":
            weighted_sum = 0.0
            for sim_idx, cluster_id in enumerate(centroids.keys()):
                weighted_sum += sims[sim_idx] * weights.get(cluster_id, 0.0)
            scores_by_label[label] = float(weighted_sum)
        else:
            scores_by_label[label] = float(np.nanmax(sims))

    return scores_by_label

def run_sense_aware_for_mode(mode: str) -> Optional[pd.DataFrame]:
    """
    Run sense-aware interpretation for one corpus mode.

    Writes per-model and primary-model outputs for the requested mode.
    In addition to the existing term-level tables, this also writes one
    per-sense similarity table so the final chapter figure can separate
    evidentiary mass from semantic composition without re-running the analysis.

    Returns the primary-model flag table for downstream comparisons, if available.
    """
    mode_tag = mode.upper()

    try:
        bert_models = load_requested_bert_models(mode_tag)
    except RuntimeError as exc:
        LOGGER.error("%s", exc)
        return None

    primary_model = preferred_bert_model_name(bert_models)
    primary_flags: Optional[pd.DataFrame] = None

    for model_name, (model, tokenizer) in bert_models.items():
        label_vecs = {
            lbl: bert_vec(model, tokenizer, lbl, is_sentence=True) for lbl in CATEGORY_LABELS
        }

        term_rows: List[Dict[str, Any]] = []
        sense_rows: List[Dict[str, Any]] = []
        sense_df = pd.DataFrame()

        for term in TARGETS:
            loaded = sense_centroids_load(term, mode=mode_tag, model_name=model_name)
            if not loaded:
                continue

            centroids, sizes = loaded
            scores = sense_aware_scores(
                centroids,
                sizes,
                label_vecs,
                strategy=SENSE_WEIGHTING,
            )
            row = {"term": term}
            row.update(scores)
            term_rows.append(row)

            ordered_cluster_ids = sorted(
                centroids.keys(),
                key=lambda cid: (-int(sizes.get(cid, 0)), int(cid)),
            )
            for cluster_id in ordered_cluster_ids:
                cluster_scores = sense_aware_scores(
                    {cluster_id: centroids[cluster_id]},
                    {cluster_id: sizes.get(cluster_id, 1)},
                    label_vecs,
                    strategy="flat",
                )
                sense_row = {
                    "term": term,
                    "sense_id": int(cluster_id),
                    "sense_size": int(sizes.get(cluster_id, 1)),
                }
                sense_row.update(cluster_scores)
                sense_rows.append(sense_row)

        if not term_rows:
            continue

        sim_df = pd.DataFrame(term_rows).set_index("term").sort_index()
        flag_df = to_flags(sim_df)

        sim_df.to_csv(RES / f"sa_similarity_main_{model_name}_{mode_tag.lower()}.csv")
        flag_df.to_csv(RES / f"sa_flags_main_{model_name}_{mode_tag.lower()}.csv")

        if sense_rows:
            sense_df = pd.DataFrame(sense_rows).sort_values(
                ["term", "sense_size", "sense_id"],
                ascending=[True, False, True],
            )
            sense_df.to_csv(
                RES / f"sa_similarity_per_sense_{model_name}_{mode_tag.lower()}.csv",
                index=False,
            )

        if model_name == primary_model:
            sim_df.to_csv(RES / f"sa_similarity_main_{mode_tag.lower()}.csv")
            flag_df.to_csv(RES / f"sa_flags_main_{mode_tag.lower()}.csv")

            if not sense_df.empty:
                sense_df.to_csv(
                    RES / f"sa_similarity_per_sense_{mode_tag.lower()}.csv",
                    index=False,
                )

            if mode_tag == "FULL":
                sim_df.to_csv(RES / "sa_similarity_main.csv")
                flag_df.to_csv(RES / "sa_flags_main.csv")
                if not sense_df.empty:
                    sense_df.to_csv(RES / "sa_similarity_per_sense.csv", index=False)

            sim_pct = rowwise_percent_scale(sim_df)
            fig_w = max(7.0, float(len(sim_df.columns)) * 1.05)
            fig_h = max(5.0, float(len(sim_df.index)) * 0.5)
            plt.figure(figsize=(fig_w, fig_h))
            ax = sns.heatmap(
                sim_pct.astype(float),
                cmap="coolwarm",
                annot=True,
                fmt=".1f",
                cbar_kws={"label": "row-wise min-max scaled contrast (0=lowest, 100=highest)"},
            )
            short_xticks = [CATEGORY_LABEL_SHORT.get(lbl, lbl) for lbl in sim_pct.columns]
            ax.set_xticklabels(short_xticks, rotation=90, fontsize=9)
            ax.set_yticklabels(sim_pct.index, fontsize=9)
            plt.title(f"Sense-aware interpretation ({mode_tag}, row-wise min-max scaled contrast)")
            plt.tight_layout()
            _save_png_svg(FIG / f"sa_similarity_main_{mode_tag.lower()}_pct.png", dpi=160)
            plt.close()

            primary_flags = flag_df

    return primary_flags

def _concept_vectors_for_bert(
    model: BertForMaskedLM,
    tokenizer: BertTokenizer,
) -> Tuple[Dict[str, Optional[np.ndarray]], pd.DataFrame]:
    """
    Build concept vectors for a BERT model by averaging anchor word embeddings
    for each concept in CONCEPT_DEFINITIONS.
    """
    concept_vecs: Dict[str, Optional[np.ndarray]] = {}
    rows: List[Dict[str, Any]] = []

    for concept_name, anchors in CONCEPT_DEFINITIONS.items():
        vecs: List[np.ndarray] = []
        for anchor in anchors:
            try:
                v = bert_vec(model, tokenizer, anchor, is_sentence=False)
            except (RuntimeError, ValueError, IndexError) as exc:
                LOGGER.warning(
                    "BERT vector failed for concept anchor '%s' in '%s': %s",
                    anchor,
                    concept_name,
                    exc,
                )
                v = None
            if v is not None and np.isfinite(v).all():
                vecs.append(v)
        n_total = len(anchors)
        n_ok = len(vecs)
        coverage = float(n_ok / n_total) if n_total > 0 else float("nan")
        rows.append(
            {
                "concept": concept_name,
                "n_anchors": int(n_total),
                "n_anchors_found": int(n_ok),
                "anchor_coverage": coverage,
            }
        )
        if vecs:
            concept_vecs[concept_name] = np.mean(np.stack(vecs, axis=0), axis=0)
        else:
            concept_vecs[concept_name] = None

    coverage_df = pd.DataFrame(rows)
    return concept_vecs, coverage_df


def _concept_vectors_for_w2v(
    model: Word2Vec,
) -> Tuple[Dict[str, Optional[np.ndarray]], pd.DataFrame]:
    """
    Build concept vectors for a Word2Vec model by averaging anchor word embeddings
    for each concept in CONCEPT_DEFINITIONS.
    """
    concept_vecs: Dict[str, Optional[np.ndarray]] = {}
    rows: List[Dict[str, Any]] = []

    for concept_name, anchors in CONCEPT_DEFINITIONS.items():
        vecs: List[np.ndarray] = []
        for anchor in anchors:
            v = w2v_vec(model, anchor)
            if v is not None and np.isfinite(v).all():
                vecs.append(v)
        n_total = len(anchors)
        n_ok = len(vecs)
        coverage = float(n_ok / n_total) if n_total > 0 else float("nan")
        rows.append(
            {
                "concept": concept_name,
                "n_anchors": int(n_total),
                "n_anchors_found": int(n_ok),
                "anchor_coverage": coverage,
            }
        )
        if vecs:
            concept_vecs[concept_name] = np.mean(np.stack(vecs, axis=0), axis=0)
        else:
            concept_vecs[concept_name] = None

    coverage_df = pd.DataFrame(rows)
    return concept_vecs, coverage_df


def _concept_alignment_table_for_bert(
    model: BertForMaskedLM,
    tokenizer: BertTokenizer,
    model_name: str,
) -> Tuple[pd.DataFrame, pd.DataFrame]:
    """
    Compute term×concept similarity table for a BERT-type model and return:

    - wide_df: rows=TARGETS, cols=concept names, values=cosine similarities
    - coverage_df: per-concept anchor coverage for this model
    """
    if not CONCEPT_DEFINITIONS:
        return pd.DataFrame(), pd.DataFrame()

    concept_vecs, coverage_df = _concept_vectors_for_bert(model, tokenizer)
    rows: List[Dict[str, Any]] = []

    for term in CONCEPT_ALIGNMENT_TARGETS:
        try:
            term_vec = bert_vec(model, tokenizer, term, is_sentence=False)
        except (RuntimeError, ValueError, IndexError) as exc:
            LOGGER.warning(
                "BERT vector failed for target '%s' in model '%s': %s",
                term,
                model_name,
                exc,
            )
            term_vec = None

        for concept_name, c_vec in concept_vecs.items():
            if term_vec is None or c_vec is None:
                sim_val = float("nan")
            else:
                sim_val = cosine(term_vec, c_vec)
            rows.append(
                {
                    "model": model_name,
                    "term": term,
                    "concept": concept_name,
                    "similarity": sim_val,
                }
            )

    long_df = pd.DataFrame(rows)
    if long_df.empty:
        return pd.DataFrame(), coverage_df
    wide_df = long_df.pivot(index="term", columns="concept", values="similarity").sort_index()
    return wide_df, coverage_df

def run_concept_alignment(mode_tag: str) -> None:
    """
    Run the bounded engawa theory-stress test for one corpus mode.

    Design:
      - boundary_edge_axis
      - connector_flow_axis
      - river_bank_probe
      - derived_hybridity = mean(boundary_edge_axis, connector_flow_axis)

    Writes:
      - per-model concept tables
      - concept_alignment_consensus_<mode>.csv
      - concept_alignment_consensus_pct_<mode>.csv
      - concept alignment heatmaps (PNG + SVG)
    """
    mode_tag = str(mode_tag).strip().upper()
    if mode_tag not in {"FULL", "CLEAN"}:
        raise ValueError("run_concept_alignment(mode_tag) expects 'FULL' or 'CLEAN'.")

    try:
        bert_models = load_requested_bert_models(mode_tag)
    except RuntimeError as exc:
        LOGGER.error("%s", exc)
        return

    per_model_tables: Dict[str, pd.DataFrame] = {}
    anchor_audit_frames: List[pd.DataFrame] = []

    for model_name, (model, tokenizer) in bert_models.items():
        concept_vecs, audit_df = _concept_vectors_for_bert(model, tokenizer)
        if not audit_df.empty:
            audit_df.insert(0, "mode", mode_tag)
            audit_df.insert(1, "model", model_name)
            anchor_audit_frames.append(audit_df)

        rows: List[Dict[str, Any]] = []
        for term in CONCEPT_ALIGNMENT_TARGETS:
            term_vec = bert_vec(model, tokenizer, term, is_sentence=False)
            for concept_name, concept_vec in concept_vecs.items():
                if term_vec is None or concept_vec is None:
                    sim_val = float("nan")
                else:
                    sim_val = cosine(term_vec, concept_vec)
                rows.append(
                    {
                        "term": term,
                        "concept": concept_name,
                        "similarity": sim_val,
                    }
                )

        if not rows:
            continue

        raw_df = pd.DataFrame(rows).pivot_table(
            index="term",
            columns="concept",
            values="similarity",
            aggfunc="mean",
        )

        required_axes = {"boundary_edge_axis", "connector_flow_axis"}
        if required_axes.issubset(set(raw_df.columns)):
            raw_df["derived_hybridity"] = raw_df[
                ["boundary_edge_axis", "connector_flow_axis"]
            ].mean(axis=1)
        else:
            raw_df["derived_hybridity"] = np.nan

        ordered_cols = [
            c for c in [
                "boundary_edge_axis",
                "connector_flow_axis",
                "derived_hybridity",
                "river_bank_probe",
            ]
            if c in raw_df.columns
        ]
        raw_df = raw_df[ordered_cols]

        raw_df.to_csv(
            ENGAWA_PROBLEM / f"concept_alignment_{model_name}_{mode_tag.lower()}.csv"
        )
        per_model_tables[model_name] = raw_df

    if anchor_audit_frames:
        pd.concat(anchor_audit_frames, ignore_index=True).to_csv(
            ENGAWA_PROBLEM / f"concept_anchor_audit_{mode_tag.lower()}.csv",
            index=False,
        )

    if not per_model_tables:
        LOGGER.warning("No concept-alignment tables were produced for mode=%s.", mode_tag)
        return

    cons = consensus(per_model_tables)
    if cons.empty:
        LOGGER.warning("Concept alignment consensus is empty for mode=%s.", mode_tag)
        return

    required_axes = {"boundary_edge_axis", "connector_flow_axis"}
    if required_axes.issubset(set(cons.columns)):
        cons["derived_hybridity"] = cons[
            ["boundary_edge_axis", "connector_flow_axis"]
        ].mean(axis=1)
    else:
        cons["derived_hybridity"] = np.nan

    ordered_cols = [
        c for c in [
            "boundary_edge_axis",
            "connector_flow_axis",
            "derived_hybridity",
            "river_bank_probe",
        ]
        if c in cons.columns
    ]
    cons = cons[ordered_cols]

    cons.to_csv(
        ENGAWA_PROBLEM / f"concept_alignment_consensus_{mode_tag.lower()}.csv"
    )

    cons_pct = rowwise_percent_scale(cons)
    cons_pct.to_csv(
        ENGAWA_PROBLEM / f"concept_alignment_consensus_pct_{mode_tag.lower()}.csv"
    )

    fig_w = max(7.0, float(len(cons.columns)) * 1.25)
    fig_h = max(4.5, float(len(cons.index)) * 0.6)
    plt.figure(figsize=(fig_w, fig_h))
    sns.heatmap(
        cons_pct.astype(float),
        cmap="coolwarm",
        annot=True,
        fmt=".1f",
        cbar_kws={"label": "row-wise min-max scaled contrast (0=lowest, 100=highest)"},
    )
    plt.title(f"Bounded engawa hypothesis-stress test ({mode_tag}, row-wise min-max scaled contrast)")
    plt.xticks(rotation=30, ha="right")
    plt.tight_layout()
    _save_png_svg(
        ENGAWA_PROBLEM_FIG / f"concept_alignment_consensus_pct_{mode_tag.lower()}.png",
        dpi=160,
    )
    plt.close()

    fig_w = max(7.0, float(len(cons.columns)) * 1.25)
    fig_h = max(4.5, float(len(cons.index)) * 0.6)
    plt.figure(figsize=(fig_w, fig_h))
    sns.heatmap(
        cons.astype(float),
        cmap="coolwarm",
        annot=True,
        fmt=".3f",
        cbar_kws={"label": "cosine similarity"},
    )
    plt.title(f"Bounded engawa hypothesis-stress test ({mode_tag}, raw cosine)")
    plt.xticks(rotation=30, ha="right")
    plt.tight_layout()
    _save_png_svg(
        ENGAWA_PROBLEM_FIG / f"concept_alignment_consensus_raw_{mode_tag.lower()}.png",
        dpi=160,
    )
    plt.close()

def consensus_target_similarity(scores_by_model: Dict[str, pd.DataFrame]) -> pd.DataFrame:
    """
    Build a target×target similarity matrix using per-target consensus over labels.

    sim(t1, t2) = cosine(consensus_scores[t1, :], consensus_scores[t2, :])
    """
    cons_df = consensus(scores_by_model)
    if cons_df.empty:
        return pd.DataFrame()
    mat = cons_df.to_numpy(dtype=float)
    n_terms = mat.shape[0]
    sims = np.zeros((n_terms, n_terms), dtype=float)
    for i in range(n_terms):
        for j in range(n_terms):
            sims[i, j] = cosine(mat[i], mat[j])
    return pd.DataFrame(sims, index=cons_df.index, columns=cons_df.index)


def plot_dendrogram(sim_mat: pd.DataFrame, outfile: Path) -> None:
    """Plot and save a hierarchical clustering dendrogram over targets."""
    if sim_mat.empty or sim_mat.shape[0] < 2:
        return

    arr = sim_mat.to_numpy(dtype=float)

    finite_mask = np.isfinite(arr).all(axis=0) & np.isfinite(arr).all(axis=1)
    arr = arr[finite_mask][:, finite_mask]
    labels = sim_mat.index[finite_mask]

    if arr.shape[0] < 2:
        return

    np.fill_diagonal(arr, 1.0)

    dist_mat = 1.0 - arr
    dist_mat[dist_mat < 0] = 0.0

    dist = squareform(dist_mat)

    linkage_mat = sch.linkage(dist, method="average")
    plt.figure(figsize=(10, 6))
    sch.dendrogram(linkage_mat, labels=list(labels), leaf_rotation=45)
    plt.title("Consensus target taxonomy (hierarchical)")
    plt.tight_layout()
    _save_png_svg(outfile, dpi=160)
    plt.close()


def plot_graph(sim_mat: pd.DataFrame, outfile: Path, thr: float = 0.35) -> None:
    """Plot and save a similarity graph over targets thresholded by edge weight."""
    if sim_mat.empty:
        return

    sim_array = sim_mat.to_numpy(dtype=float)
    terms = list(sim_mat.index)
    n_terms = len(terms)

    graph = nx.Graph()
    for i, term_a in enumerate(terms):
        graph.add_node(term_a)
        for j in range(i + 1, n_terms):
            term_b = terms[j]
            sim_val = float(sim_array[i, j])
            if not np.isnan(sim_val) and sim_val >= thr:
                graph.add_edge(term_a, term_b, weight=sim_val)

    if graph.number_of_nodes() == 0:
        return

    pos = nx.spring_layout(
        graph,
        seed=SEED,
        k=1.0 / math.sqrt(max(graph.number_of_nodes(), 1)),
    )
    plt.figure(figsize=(9, 8))

    # edges
    for node_u, node_v in graph.edges():
        x_values = [pos[node_u][0], pos[node_v][0]]
        y_values = [pos[node_u][1], pos[node_v][1]]
        plt.plot(x_values, y_values, linewidth=1.5)

    # nodes
    xs = [pos[node][0] for node in graph.nodes()]
    ys = [pos[node][1] for node in graph.nodes()]
    plt.scatter(xs, ys, s=1200)

    # labels
    for node, (x_coord, y_coord) in pos.items():
        plt.text(x_coord, y_coord, str(node), ha="center", va="center")

    plt.title("Consensus target similarity graph")
    plt.axis("off")
    plt.tight_layout()
    _save_png_svg(outfile, dpi=160)
    plt.close()

def build_summary_dashboard() -> None:
    """
    Build a compact Paper 4 dashboard for screening and interpretation.

    The dashboard is term-centred and keeps FULL/CLEAN audit logic visible.
    It combines:
      - polysemy sufficiency and structure under FULL and CLEAN
      - architecture-facing classification under FULL and CLEAN
      - sense-aware interpretation under FULL and CLEAN
      - zero-shot bootstrap summary, when available
      - bounded engawa theory-stress test summary

    Writes results_paper4/summary_dashboard.csv.
    """
    primary_model = "bert_ft" if USE_BERT_FINETUNED else "bert_base"

    def _load_indexed_csv(path: Path, index_col: str = "term") -> pd.DataFrame:
        if not path.is_file():
            return pd.DataFrame()
        df = pd.read_csv(path)
        if df.empty:
            return df
        if index_col in df.columns:
            df = df.drop_duplicates(subset=[index_col]).set_index(index_col)
        return df

    def _load_wide_csv(path: Path) -> pd.DataFrame:
        return pd.read_csv(path, index_col=0) if path.is_file() else pd.DataFrame()

    def _top_item(sim_table: pd.DataFrame, term_name: str) -> Tuple[str, float]:
        if sim_table.empty or term_name not in sim_table.index:
            return "", float("nan")
        sim_row = sim_table.loc[[term_name]].iloc[0]
        sim_row = sim_row.apply(lambda value: pd.to_numeric(value, errors="coerce"))
        sim_row = sim_row.dropna().sort_values(ascending=False)
        if sim_row.empty:
            return "", float("nan")
        return str(sim_row.index[0]), float(sim_row.iloc[0])

    def _labels_on(flag_table: pd.DataFrame, term_name: str) -> str:
        if flag_table.empty or term_name not in flag_table.index:
            return ""
        flag_row = flag_table.loc[[term_name]].iloc[0]
        flag_row = flag_row.apply(lambda value: pd.to_numeric(value, errors="coerce")).fillna(0).astype(int)
        return "; ".join([str(label) for label in flag_row.index[flag_row == 1].tolist()])

    poly_full = _load_indexed_csv(RES / f"polysemy_summary_{primary_model}_full.csv")
    poly_clean = _load_indexed_csv(RES / f"polysemy_summary_{primary_model}_clean.csv")

                    not concept_cons_full.empty and term in concept_cons_full.index) else pd.DataFrame()
        clean_concepts = concept_cons_clean if (
                    not concept_cons_clean.empty and term in concept_cons_clean.index) else pd.DataFrame()

        if not full_concepts.empty:
            concept_row_full = concept_cons_full.loc[[term]].iloc[0]
            sims_full = concept_row_full.apply(lambda value: pd.to_numeric(value, errors="coerce"))
            sims_full = sims_full.dropna().sort_values(ascending=False)
            if not sims_full.empty:
                row["engawa_full_concept_top_name"] = str(sims_full.index[0])
                row["engawa_full_concept_top_sim"] = float(sims_full.iloc[0])
            else:
                row["engawa_full_concept_top_name"] = ""
                row["engawa_full_concept_top_sim"] = float("nan")

            for cname in ("boundary_edge_axis", "connector_flow_axis", "derived_hybridity", "river_bank_probe"):
                row[f"engawa_full_{cname}_sim"] = float(
                    pd.to_numeric(concept_cons_full.at[term, cname], errors="coerce")
                ) if cname in concept_cons_full.columns else float("nan")
        else:
            row["engawa_full_concept_top_name"] = ""
            row["engawa_full_concept_top_sim"] = float("nan")
            for cname in ("boundary_edge_axis", "connector_flow_axis", "derived_hybridity", "river_bank_probe"):
                row[f"engawa_full_{cname}_sim"] = float("nan")

        if not clean_concepts.empty:
            concept_row_clean = concept_cons_clean.loc[[term]].iloc[0]
            sims_clean = concept_row_clean.apply(lambda value: pd.to_numeric(value, errors="coerce"))
            sims_clean = sims_clean.dropna().sort_values(ascending=False)
            if not sims_clean.empty:
                row["engawa_clean_concept_top_name"] = str(sims_clean.index[0])
                row["engawa_clean_concept_top_sim"] = float(sims_clean.iloc[0])
            else:
                row["engawa_clean_concept_top_name"] = ""
                row["engawa_clean_concept_top_sim"] = float("nan")

            for cname in ("boundary_edge_axis", "connector_flow_axis", "derived_hybridity", "river_bank_probe"):
                row[f"engawa_clean_{cname}_sim"] = float(
                    pd.to_numeric(concept_cons_clean.at[term, cname], errors="coerce")
                ) if cname in concept_cons_clean.columns else float("nan")
        else:
            row["engawa_clean_concept_top_name"] = ""
            row["engawa_clean_concept_top_sim"] = float("nan")
            for cname in ("boundary_edge_axis", "connector_flow_axis", "derived_hybridity", "river_bank_probe"):
                row[f"engawa_clean_{cname}_sim"] = float("nan")

        full_ok = np.isfinite(row["poly_full_contexts"]) and row["poly_full_contexts"] >= POLY_MIN_INST
        clean_ok = np.isfinite(row["poly_clean_contexts"]) and row["poly_clean_contexts"] >= POLY_MIN_INST
        row["screening_has_full_contexts"] = bool(full_ok)
        row["screening_has_clean_contexts"] = bool(clean_ok)
        row["screening_ready_for_interpretation"] = bool(full_ok and clean_ok)

        full_k = row["poly_full_best_k"]
        clean_k = row["poly_clean_best_k"]
        if not full_ok or not clean_ok:
            row["screening_note"] = "insufficient_contexts"
        elif (
            np.isfinite(full_k)
            and np.isfinite(clean_k)
            and float(full_k) <= 1.0
            and float(clean_k) <= 1.0
        ):
            row["screening_note"] = "weak_internal_structure"
        else:
            row["screening_note"] = "proceed_to_interpretive_screening"

        dashboard_rows.append(row)

    if dashboard_rows:
        dash_df = pd.DataFrame(dashboard_rows)
        dash_path = RES / "summary_dashboard.csv"
        dash_df.to_csv(dash_path, index=False)
        LOGGER.info("Wrote compact dashboard summary to %s", dash_path)
    else:
        LOGGER.info("No dashboard rows created; summary_dashboard.csv not written")

def build_screening_decision_table() -> None:
    """
    Build an explicit Paper 4 screening decision table from summary_dashboard.csv.

    This makes the eight-term screening outcome auditable and easy to cite
    in the write-up.

    Writes:
        results_paper4/screening_decision_table.csv
    """
    dash_path = RES / "summary_dashboard.csv"
    if not dash_path.is_file():
        LOGGER.info("screening_decision_table.csv not built because summary_dashboard.csv is missing")
        return

    dash_df = pd.read_csv(dash_path)
    if dash_df.empty or "term" not in dash_df.columns:
        LOGGER.info("screening_decision_table.csv not built because summary_dashboard.csv is empty")
        return

    def _to_bool(value: Any) -> bool:
        if isinstance(value, bool):
            return value
        if pd.isna(value):
            return False
        return str(value).strip().lower() in {"1", "true", "yes", "y"}

    rows: List[Dict[str, Any]] = []

    for _, rec in dash_df.iterrows():
        term = str(rec.get("term", "")).strip()
        if not term:
            continue

        full_ctx = float(pd.to_numeric(rec.get("poly_full_contexts"), errors="coerce"))
        clean_ctx = float(pd.to_numeric(rec.get("poly_clean_contexts"), errors="coerce"))
        full_k = float(pd.to_numeric(rec.get("poly_full_best_k"), errors="coerce"))
        clean_k = float(pd.to_numeric(rec.get("poly_clean_best_k"), errors="coerce"))
        full_stability = float(pd.to_numeric(rec.get("poly_full_stability_ari_median"), errors="coerce"))
        clean_stability = float(pd.to_numeric(rec.get("poly_clean_stability_ari_median"), errors="coerce"))
        full_span = float(pd.to_numeric(rec.get("poly_full_span_match_rate"), errors="coerce"))
        clean_span = float(pd.to_numeric(rec.get("poly_clean_span_match_rate"), errors="coerce"))
        full_perm = float(pd.to_numeric(rec.get("poly_full_perm_p_value"), errors="coerce"))
        clean_perm = float(pd.to_numeric(rec.get("poly_clean_perm_p_value"), errors="coerce"))
        full_matched = float(pd.to_numeric(rec.get("poly_full_matched_span_vectors"), errors="coerce"))
        clean_matched = float(pd.to_numeric(rec.get("poly_clean_matched_span_vectors"), errors="coerce"))
        full_fallback = float(pd.to_numeric(rec.get("poly_full_sentence_mean_fallback_vectors"), errors="coerce"))
        clean_fallback = float(pd.to_numeric(rec.get("poly_clean_sentence_mean_fallback_vectors"), errors="coerce"))

        full_fallback_ratio = (
            full_fallback / (full_matched + full_fallback)
            if np.isfinite(full_matched) and np.isfinite(full_fallback) and (full_matched + full_fallback) > 0.0
            else float("nan")
        )
        clean_fallback_ratio = (
            clean_fallback / (clean_matched + clean_fallback)
            if np.isfinite(clean_matched) and np.isfinite(clean_fallback) and (clean_matched + clean_fallback) > 0.0
            else float("nan")
        )

        term_name = str(rec.get("term", ""))
        term_override = SCREEN_TERM_OVERRIDES.get(term_name, {})
        term_max_fallback_ratio = float(term_override.get("max_fallback_ratio", SCREEN_MAX_FALLBACK_RATIO))

        has_full = np.isfinite(full_ctx) and full_ctx >= SCREEN_MIN_FULL_CONTEXTS
        has_clean = np.isfinite(clean_ctx) and clean_ctx >= SCREEN_MIN_CLEAN_CONTEXTS

        full_polysemous = np.isfinite(full_k) and full_k > 1.0
        clean_polysemous = np.isfinite(clean_k) and clean_k > 1.0

        full_stable = (not full_polysemous) or (
            np.isfinite(full_stability) and full_stability >= SCREEN_MIN_STABILITY_IF_POLYSEMOUS
        )
        clean_stable = (not clean_polysemous) or (
            np.isfinite(clean_stability) and clean_stability >= SCREEN_MIN_STABILITY_IF_POLYSEMOUS
        )
        full_perm_ok = (not full_polysemous) or (
            np.isfinite(full_perm) and full_perm <= SCREEN_MAX_PERM_P_IF_POLYSEMOUS
        )
        clean_perm_ok = (not clean_polysemous) or (
            np.isfinite(clean_perm) and clean_perm <= SCREEN_MAX_PERM_P_IF_POLYSEMOUS
        )
        full_span_ok = np.isfinite(full_span) and full_span >= SCREEN_MIN_SPAN_MATCH_RATE
        clean_span_ok = np.isfinite(clean_span) and clean_span >= SCREEN_MIN_SPAN_MATCH_RATE

        full_fallback_ok = (not np.isfinite(full_fallback_ratio)) or (full_fallback_ratio <= term_max_fallback_ratio)
        clean_fallback_ok = (not np.isfinite(clean_fallback_ratio)) or (clean_fallback_ratio <= term_max_fallback_ratio)

        k_shift_abs = (
            abs(int(full_k) - int(clean_k))
            if np.isfinite(full_k) and np.isfinite(clean_k)
            else 999
        )
        k_shift_ok = k_shift_abs <= SCREEN_MAX_FULL_CLEAN_K_SHIFT

        if not has_full or not has_clean:
            auto_decision = "reject_insufficient_contexts"
            decision_reason = "insufficient_contexts"
        elif not full_span_ok or not clean_span_ok:
            auto_decision = "hold_low_span_match"
            decision_reason = "low_span_match_rate"
        elif not full_fallback_ok or not clean_fallback_ok:
            auto_decision = "hold_high_fallback_ratio"
            decision_reason = "extraction_quality_too_noisy"
        elif not k_shift_ok:
            auto_decision = "appendix_full_clean_instability"
            decision_reason = "large_full_clean_k_shift"
        elif SCREEN_REQUIRE_CLEAN_NONCOLLAPSE and full_polysemous and not clean_polysemous:
            auto_decision = "appendix_clean_collapse"
            decision_reason = "structure_collapses_in_clean"
        elif not full_stable or not clean_stable:
            auto_decision = "hold_low_stability"
            decision_reason = "low_cluster_stability"
        elif full_polysemous or clean_polysemous:
            auto_decision = "retain_for_main_body_comparison"
            if full_polysemous and clean_polysemous and not full_perm_ok and not clean_perm_ok:
                decision_reason = "recoverable_internal_structure_with_weak_separation_warning"
            elif (full_polysemous and not full_perm_ok) or (clean_polysemous and not clean_perm_ok):
                decision_reason = "recoverable_internal_structure_with_one_corpus_separation_warning"
            else:
                decision_reason = "recoverable_internal_structure"
        else:
            auto_decision = "retain_but_treat_as_weaker_case"
            decision_reason = "interpretable_but_weak_internal_structure"

        rows.append(
            {
                "term": term,
                "poly_full_contexts": full_ctx,
                "poly_clean_contexts": clean_ctx,
                "poly_full_best_k": full_k,
                "poly_clean_best_k": clean_k,
                "poly_full_stability_ari_median": full_stability,
                "poly_clean_stability_ari_median": clean_stability,
                "poly_full_span_match_rate": full_span,
                "poly_clean_span_match_rate": clean_span,
                "screening_has_full_contexts": bool(has_full),
                "screening_has_clean_contexts": bool(has_clean),
                "screening_full_span_ok": bool(full_span_ok),
                "screening_clean_span_ok": bool(clean_span_ok),
                "screening_full_stable_enough": bool(full_stable),
                "screening_clean_stable_enough": bool(clean_stable),
                "poly_full_perm_p_value": full_perm,
                "poly_clean_perm_p_value": clean_perm,
                "screening_full_perm_ok": bool(full_perm_ok),
                "screening_clean_perm_ok": bool(clean_perm_ok),
                "screening_ready_for_interpretation": bool(
                    _to_bool(rec.get("screening_ready_for_interpretation"))
                ),
                "screening_note_from_dashboard": str(rec.get("screening_note", "")).strip(),
                "auto_decision": auto_decision,
                "decision_reason": decision_reason,
                "manual_priority": SCREEN_MANUAL_PRIORITY.get(term, ""),
                "architecture_note": SCREEN_ARCHITECTURE_NOTE.get(term, ""),
            }
        )

    out_df = pd.DataFrame(rows)
    out_path = RES / "screening_decision_table.csv"
    out_df.to_csv(out_path, index=False)
    LOGGER.info("Wrote explicit screening decision table to %s", out_path)

def _read_csv_optional(path: Path, index_col: Optional[str] = None) -> pd.DataFrame:
    if not path.is_file():
        return pd.DataFrame()
    df = pd.read_csv(path)
    if df.empty:
        return df
    if index_col is not None and index_col in df.columns:
        df = df.set_index(index_col)
    return df

def _to_float_scalar(value: Any, default: float = float("nan")) -> float:
    if value is None:
        return default
    try:
        if pd.isna(value):
            return default
    except TypeError:
        pass
    try:
        return float(value)
    except (TypeError, ValueError):
        return default


def _to_int_scalar(value: Any, default: int = 0) -> int:
    val = _to_float_scalar(value, default=float(default))
    if not np.isfinite(val):
        return default
    return int(round(val))

def _truncate_sentence(text: str, max_chars: int = 96) -> str:
    clean = re.sub(r"\s+", " ", str(text)).strip()
    if len(clean) <= max_chars:
        return clean
    return clean[: max_chars - 1].rstrip() + "…"

def _norm_label(text: str) -> str:
    """Normalise a label for tolerant register matching."""
    work = str(text).strip().lower()
    work = work.replace("&", " and ")
    work = re.sub(r"[_\-/]+", " ", work)
    work = re.sub(r"\s+", " ", work)
    return work.strip()


def _match_main_register_columns(columns: List[str]) -> Dict[str, List[str]]:
    """
    Resolve exported similarity-table columns onto the five Paper 4 main registers.

    Exact and near-exact matches are preferred. Multiple matched columns may contribute
    to one main register, but a single source column is still assigned to only one
    register to avoid double counting.
    """
    matched: Dict[str, List[str]] = {reg: [] for reg in MAIN_REGISTER_ORDER}
    used_cols: set[str] = set()

    # Pass 1: exact normalised match, but allow more than one exact column per register.
    for register_name in MAIN_REGISTER_ORDER:
        alias_norms = {_norm_label(alias) for alias in MAIN_REGISTER_ALIASES.get(register_name, [])}
        for col_name in columns:
            if col_name in used_cols:
                continue
            col_norm = _norm_label(col_name)
            if col_norm in alias_norms:
                matched[register_name].append(col_name)
                used_cols.add(col_name)

    # Pass 2: contains / startswith / endswith match, again allowing multiple columns.
    for register_name in MAIN_REGISTER_ORDER:
        alias_norms = [_norm_label(alias) for alias in MAIN_REGISTER_ALIASES.get(register_name, [])]
        for col_name in columns:
            if col_name in used_cols:
                continue
            col_norm = _norm_label(col_name)

            if any(
                col_norm == alias_norm
                or col_norm.startswith(alias_norm + " ")
                or col_norm.endswith(" " + alias_norm)
                or f" {alias_norm} " in f" {col_norm} "
                for alias_norm in alias_norms
            ):
                matched[register_name].append(col_name)
                used_cols.add(col_name)

    return matched

def _screening_min_stability_lookup() -> Dict[str, float]:
    """Return min(FULL, CLEAN) stability per term from the screening decision table."""
    screen_df = _read_csv_optional(RES / "screening_decision_table.csv")
    if screen_df.empty or "term" not in screen_df.columns:
        return {}

    lookup: Dict[str, float] = {}

    for _, rec in screen_df.iterrows():
        term = str(rec.get("term", "")).strip()
        if not term:
            continue

        full_stab = _to_float_scalar(rec.get("poly_full_stability_ari_median"))
        clean_stab = _to_float_scalar(rec.get("poly_clean_stability_ari_median"))
        finite_vals = [v for v in (full_stab, clean_stab) if np.isfinite(v)]

        lookup[term] = min(finite_vals) if finite_vals else float("nan")

    return lookup


def _derive_mixed_from_base_registers(
    reg_df: pd.DataFrame,
    term_labels: Optional[pd.Series] = None,
) -> pd.DataFrame:
    """
    Derive 'mixed' from concurrent strength in the four base registers rather than
    treating it as a primitive competitor label.

    mixed = weaker of the two strongest active base registers.
    Screening reliability is handled in the screening table, not inside the register profile.
    """
    if reg_df.empty:
        return reg_df

    out = reg_df.copy()
    base_registers = ["material", "conceptual", "virtual", "mediating"]

    if term_labels is None:
        term_series = pd.Series(out.index.astype(str), index=out.index, dtype=object)
    else:
        term_series = pd.Series(term_labels.astype(str).tolist(), index=out.index, dtype=object)

    base_frame = out.loc[:, base_registers].apply(lambda col: pd.to_numeric(col, errors="coerce"))

    for row_idx, row_series in base_frame.iterrows():
        vals = row_series.reindex(base_registers).to_numpy(dtype=float)
        finite = np.sort(vals[np.isfinite(vals)])

        if finite.size < 2:
            mixed_raw = float("nan")
        else:
            mixed_raw = float(finite[-2])

        out.at[row_idx, "mixed"] = mixed_raw

    return out


def _aggregate_to_main_registers(
    sim_df: pd.DataFrame,
    term_labels: Optional[pd.Series] = None,
) -> pd.DataFrame:
    """
    Collapse similarity tables onto the fixed five Paper 4 main registers.

    The four base registers are aggregated directly from the zero-shot outputs.
    'mixed' is then derived from concurrent strength across those base registers,
    rather than taken from any direct prompt label.
    """
    if sim_df.empty:
        return pd.DataFrame(index=sim_df.index, columns=MAIN_REGISTER_ORDER, dtype=float)

    numeric = sim_df.apply(lambda col: pd.to_numeric(col, errors="coerce"))
    matched_cols = _match_main_register_columns([str(col) for col in numeric.columns])

    base_registers = [reg for reg in MAIN_REGISTER_ORDER if reg != "mixed"]
    out: Dict[str, pd.Series] = {}

    for register_name in base_registers:
        reg_cols = [col for col in matched_cols.get(register_name, []) if col in numeric.columns]
        if reg_cols:
            out[register_name] = numeric[reg_cols].mean(axis=1, skipna=True)
        else:
            out[register_name] = pd.Series(np.nan, index=numeric.index, dtype=float)

    out_df = pd.DataFrame(out, index=numeric.index)
    out_df["mixed"] = np.nan
    out_df = _derive_mixed_from_base_registers(out_df, term_labels=term_labels)

    missing_regs = [reg for reg in base_registers if out_df[reg].isna().all()]
    if missing_regs:
        LOGGER.warning(
            "Main-register aggregation found no matching columns for: %s | source columns=%s",
            ", ".join(missing_regs),
            ", ".join([str(col) for col in numeric.columns]),
        )

    return out_df.loc[:, MAIN_REGISTER_ORDER]

def _scale_full_clean_pairs(
    full_df: pd.DataFrame,
    clean_df: pd.DataFrame,
) -> Tuple[pd.DataFrame, pd.DataFrame]:
    combined_index: List[Any] = sorted(
        set(full_df.index.tolist()) | set(clean_df.index.tolist())
    )
    full_work = full_df.reindex(combined_index)
    clean_work = clean_df.reindex(combined_index)
    scaled_full = pd.DataFrame(index=combined_index, columns=MAIN_REGISTER_ORDER, dtype=float)
    scaled_clean = pd.DataFrame(index=combined_index, columns=MAIN_REGISTER_ORDER, dtype=float)

    for term in combined_index:
        full_row = cast(pd.Series, full_work.loc[term])
        clean_row = cast(pd.Series, clean_work.loc[term])

        full_vals = np.asarray(pd.Series(full_row), dtype=float)
        clean_vals = np.asarray(pd.Series(clean_row), dtype=float)

        combo = np.concatenate([full_vals, clean_vals])
        scaled = np.full(combo.shape, np.nan, dtype=float)
        valid = np.isfinite(combo)

        if np.any(valid):
            valid_vals = combo[valid]
            lo = float(valid_vals.min())
            hi = float(valid_vals.max())
            if hi > lo:
                scaled[valid] = 100.0 * (combo[valid] - lo) / (hi - lo)
            else:
                scaled[valid] = 100.0

        scaled_full.loc[term] = scaled[: len(MAIN_REGISTER_ORDER)]
        scaled_clean.loc[term] = scaled[len(MAIN_REGISTER_ORDER):]

    return scaled_full, scaled_clean


def _row_to_composition(row: pd.Series) -> np.ndarray:
    vals = pd.to_numeric(row, errors="coerce").to_numpy(dtype=float)
    out = np.zeros_like(vals, dtype=float)
    valid = np.isfinite(vals)
    if not np.any(valid):
        return out
    work = vals[valid]
    work -= float(np.nanmin(work))
    if float(work.sum()) <= 0.0:
        work = np.ones_like(work, dtype=float)
    out[valid] = 100.0 * work / float(work.sum())
    return out


def _ordered_cluster_ids(
    sizes: Dict[int, int],
    centroids: Dict[int, np.ndarray],
) -> List[int]:
    pool = sorted(set(sizes.keys()) | set(centroids.keys()))
    return sorted(pool, key=lambda cid: (-int(sizes.get(cid, 0)), int(cid)))


def _aligned_polysemy_slots(
    term: str,
    model_name: str,
) -> List[Tuple[Optional[int], Optional[int]]]:
    full_loaded = sense_centroids_load(term, mode="FULL", model_name=model_name)
    clean_loaded = sense_centroids_load(term, mode="CLEAN", model_name=model_name)

    full_centroids, full_sizes = full_loaded if full_loaded else ({}, {})
    clean_centroids, clean_sizes = clean_loaded if clean_loaded else ({}, {})

    slots: List[Tuple[Optional[int], Optional[int]]] = []
    full_order = _ordered_cluster_ids(full_sizes, full_centroids)
    clean_unused = set(_ordered_cluster_ids(clean_sizes, clean_centroids))

    for full_cluster_id in full_order:
        matched_clean_id: Optional[int] = None
        best_sim = -np.inf

        if full_cluster_id in full_centroids:
            for clean_cluster_id in list(clean_unused):
                sim_val = cosine(full_centroids[full_cluster_id], clean_centroids[clean_cluster_id])
                sim_score = float(sim_val) if np.isfinite(sim_val) else -np.inf
                if sim_score > best_sim:
                    best_sim = sim_score
                    matched_clean_id = clean_cluster_id

        if matched_clean_id is not None:
            clean_unused.remove(matched_clean_id)
        slots.append((full_cluster_id, matched_clean_id))

    for clean_cluster_id in _ordered_cluster_ids(clean_sizes, clean_centroids):
        if clean_cluster_id in clean_unused:
            slots.append((None, clean_cluster_id))

    return slots


def _load_polysemy_representatives(
    term: str,
    mode: str,
    model_name: str,
) -> pd.DataFrame:
    path = RES / f"polysemy_representatives_{term}_{model_name}_{mode.lower()}.csv"
    if not path.is_file():
        return pd.DataFrame(columns=["cluster", "rank", "distance", "sentence"])
    df = pd.read_csv(path)
    if df.empty:
        return df
    return df.sort_values(["cluster", "rank"])


def _cluster_snippet(rep_df: pd.DataFrame, cluster_id: Optional[int]) -> str:
    if cluster_id is None or rep_df.empty or "cluster" not in rep_df.columns:
        return "—"
    sub = rep_df.loc[rep_df["cluster"].astype(int).eq(int(cluster_id))].copy()
    if sub.empty:
        return "—"
    sent = str(sub.iloc[0].get("sentence", "")).strip()
    return _truncate_sentence(sent)


def plot_screening_audit_table(out_path: Path) -> bool:
    """Render the chapter-facing screening audit table."""
    screen_df = _read_csv_optional(RES / "screening_decision_table.csv")
    if screen_df.empty:
        return False

    cols = [
        "term",
        "poly_full_contexts",
        "poly_clean_contexts",
        "poly_full_best_k",
        "poly_clean_best_k",
        "poly_full_span_match_rate",
        "poly_clean_span_match_rate",
        "poly_full_stability_ari_median",
        "poly_clean_stability_ari_median",
        "auto_decision",
    ]
    work = screen_df.loc[:, [c for c in cols if c in screen_df.columns]].copy()
    if work.empty:
        return False

    rename_map = {
        "term": "term",
        "poly_full_contexts": "FULL n",
        "poly_clean_contexts": "CLEAN n",
        "poly_full_best_k": "FULL k",
        "poly_clean_best_k": "CLEAN k",
        "poly_full_span_match_rate": "FULL span",
        "poly_clean_span_match_rate": "CLEAN span",
        "poly_full_stability_ari_median": "FULL stab",
        "poly_clean_stability_ari_median": "CLEAN stab",
        "auto_decision": "decision",
    }
    work = work.rename(columns=rename_map)

    for col in ("FULL n", "CLEAN n", "FULL k", "CLEAN k"):
        if col in work.columns:
            work[col] = pd.to_numeric(work[col], errors="coerce").map(
                lambda v: "" if pd.isna(v) else f"{int(round(float(v)))}"
            )

    for col in ("FULL span", "CLEAN span", "FULL stab", "CLEAN stab"):
        if col in work.columns:
            work[col] = pd.to_numeric(work[col], errors="coerce").map(
                lambda v: "" if pd.isna(v) else f"{float(v):.2f}"
            )

    if "decision" in work.columns:
        work["decision"] = (
            work["decision"]
            .astype(str)
            .str.replace("_", " ", regex=False)
            .str.strip()
            .map(lambda s: _wrap_text_block(s, width=22))
        )

    def _decision_fill(text: str) -> str:
        norm = str(text).replace("\n", " ").strip().lower()
        if "retain" in norm:
            return "#eaf4ea"
        if "hold" in norm:
            return "#fbf1d9"
        return "white"

    fig = plt.figure(figsize=(15.0, max(4.4, 1.7 + float(len(work)) * 0.66)))
    ax = cast(Axes, fig.add_subplot(111))
    ax.axis("off")

    table = mpl_table(
        ax,
        cellText=work.values.tolist(),
        colLabels=work.columns.tolist(),
        cellLoc="center",
        loc="center",
    )
    table.auto_set_font_size(False)
    table.set_fontsize(8.4)
    table.scale(1.0, 1.7)

    col_widths = {
        "term": 0.10,
        "FULL n": 0.09,
        "CLEAN n": 0.09,
        "FULL k": 0.08,
        "CLEAN k": 0.08,
        "FULL span": 0.10,
        "CLEAN span": 0.10,
        "FULL stab": 0.10,
        "CLEAN stab": 0.10,
        "decision": 0.21,
    }

    for (row_idx, col_idx), cell in table.get_celld().items():
        cell.set_linewidth(0.6)

        if row_idx == 0:
            cell.set_facecolor("#ececec")
            cell.set_text_props(weight="bold")
            cell.set_height(cell.get_height() * 1.18)
        else:
            decision_text = str(work.iloc[row_idx - 1].get("decision", ""))
            cell.set_facecolor(_decision_fill(decision_text))

        if col_idx < len(work.columns):
            col_name = str(work.columns[col_idx])
            if col_name in col_widths:
                cell.set_width(col_widths[col_name])

            if row_idx > 0 and col_name in {"term", "decision"}:
                cell.set_text_props(ha="left")
                cell.PAD = 0.02

    ax.set_title("Table 1. Screening audit under FULL and CLEAN", fontsize=12, pad=12)
    fig.text(
        0.5,
        0.028,
        "n = matched contextual instances, k = selected number of recovered contextual clusters, span = proportion of candidate sentences with an explicit target-span match, stab = median bootstrap ARI stability. Green rows are retain decisions, pale amber rows are hold decisions.",
        ha="center",
        va="bottom",
        fontsize=8.2,
    )
    plt.tight_layout(rect=(0.0, 0.06, 1.0, 1.0))
    _save_png_svg(out_path, fig=fig, dpi=180)
    plt.close(fig)
    return True

def plot_polysemy_plate(out_path: Path) -> bool:
    """Render the chapter-facing contextual cluster recovery figure with within-term FULL/CLEAN alignment."""
    model_name = _primary_model_key()

    term_payloads: List[Dict[str, Any]] = []
    max_total = 0
    max_slots = 0

    def _short_cue(rep_df: pd.DataFrame, cluster_id: Optional[int]) -> str:
        cue = _cluster_snippet(rep_df, cluster_id)
        if cue == "—":
            return "no representative cue available"
        return _truncate_sentence(cue, max_chars=52)

    for term in TARGETS:
        full_loaded = sense_centroids_load(term, model_name=model_name)
        clean_loaded = sense_centroids_load(term, mode="CLEAN", model_name=model_name)

        full_centroids, full_sizes = full_loaded if full_loaded else ({}, {})
        clean_centroids, clean_sizes = clean_loaded if clean_loaded else ({}, {})

        slots = _aligned_polysemy_slots(term, model_name=model_name)
        full_reps = _load_polysemy_representatives(term, "FULL", model_name)
        clean_reps = _load_polysemy_representatives(term, "CLEAN", model_name)

        full_total = int(sum(full_sizes.values())) if full_sizes else 0
        clean_total = int(sum(clean_sizes.values())) if clean_sizes else 0
        max_total = max(max_total, full_total, clean_total)
        max_slots = max(max_slots, len(slots))

        term_payloads.append(
            {
                "term": term,
                "slots": slots,
                "full_sizes": full_sizes,
                "clean_sizes": clean_sizes,
                "full_reps": full_reps,
                "clean_reps": clean_reps,
                "full_total": full_total,
                "clean_total": clean_total,
            }
        )

    if not term_payloads or max_total <= 0:
        return False

    fig = plt.figure(figsize=(16.8, max(9.8, float(len(term_payloads)) * 2.25)))
    gs = fig.add_gridspec(
        nrows=len(term_payloads),
        ncols=2,
        width_ratios=[1.28, 0.92],
        hspace=0.48,
        wspace=0.10,
    )

    polysemy_cmap = plt.get_cmap("tab20")
    color_bank = [
        polysemy_cmap(pos)
        for pos in np.linspace(0.0, 1.0, num=20, endpoint=False)
    ]

    for row_idx, payload in enumerate(term_payloads):
        ax_bar = fig.add_subplot(gs[row_idx, 0])
        ax_txt = fig.add_subplot(gs[row_idx, 1])

        slots = payload["slots"]
        full_sizes = payload["full_sizes"]
        clean_sizes = payload["clean_sizes"]

        full_left = 0.0
        clean_left = 0.0

        for slot_idx, (full_cluster_id, clean_cluster_id) in enumerate(slots, start=1):
            color = color_bank[(slot_idx - 1) % len(color_bank)]
            full_width = float(full_sizes.get(full_cluster_id, 0)) if full_cluster_id is not None else 0.0
            clean_width = float(clean_sizes.get(clean_cluster_id, 0)) if clean_cluster_id is not None else 0.0

            if full_width > 0:
                ax_bar.barh(
                    1.0,
                    full_width,
                    left=full_left,
                    height=0.34,
                    color=color,
                    edgecolor="white",
                )
                full_left += full_width

            if clean_width > 0:
                ax_bar.barh(
                    0.0,
                    clean_width,
                    left=clean_left,
                    height=0.34,
                    color=color,
                    edgecolor="white",
                )
                clean_left += clean_width

        ax_bar.set_xlim(0.0, max_total * 1.12)
        ax_bar.set_yticks([1.0, 0.0])
        ax_bar.set_yticklabels(["FULL", "CLEAN"], fontsize=8.5)
        ax_bar.set_title(str(payload["term"]), loc="left", fontsize=11, weight="bold")
        ax_bar.grid(axis="x", alpha=0.25)

        if row_idx == len(term_payloads) - 1:
            ax_bar.set_xlabel("matched contextual instances", fontsize=9)

        ax_bar.text(
            payload["full_total"] + max_total * 0.015,
            1.0,
            f"n={payload['full_total']}",
            va="center",
            fontsize=8,
        )
        ax_bar.text(
            payload["clean_total"] + max_total * 0.015,
            0.0,
            f"n={payload['clean_total']}",
            va="center",
            fontsize=8,
        )

        ax_txt.axis("off")
        text_lines: List[str] = []

        for slot_idx, (full_cluster_id, clean_cluster_id) in enumerate(slots, start=1):
            full_cue = _short_cue(payload["full_reps"], full_cluster_id)
            clean_cue = _short_cue(payload["clean_reps"], clean_cluster_id)

            text_lines.append(f"C{slot_idx}  FULL: {full_cue}")
            text_lines.append(f"    CLEAN: {clean_cue}")

        ax_txt.text(
            0.0,
            1.0,
            "\n".join([_wrap_text_block(line, width=54) for line in text_lines]),
            va="top",
            ha="left",
            fontsize=8.0,
            linespacing=1.25,
        )

    legend_slots = min(max_slots, 4)
    legend_handles = [
        Line2D([0], [0], color=color_bank[(slot_idx - 1) % len(color_bank)], linewidth=8, label=f"C{slot_idx}")
        for slot_idx in range(1, legend_slots + 1)
    ]

    fig.suptitle(
        "Figure 2. Contextual cluster recovery across target terms",
        fontsize=13,
        y=0.988,
    )
    if legend_handles:
        fig.legend(
            handles=legend_handles,
            loc="upper center",
            ncol=max(1, legend_slots),
            frameon=False,
            bbox_to_anchor=(0.5, 0.958),
        )
    fig.text(
        0.5,
        0.012,
        "Bar length indicates matched-context count. Colored segments represent recovered contextual clusters aligned within each target term across FULL and CLEAN. C1, C2, and so on are aligned cluster slots within each term only; the cue text gives one short representative sentence for each slot.",
        ha="center",
        va="bottom",
        fontsize=8.6,
    )
    plt.tight_layout(rect=(0.0, 0.05, 1.0, 0.90))
    _save_png_svg(out_path, fig=fig, dpi=180)
    plt.close(fig)
    return True

def _aggregate_sense_rows_to_registers(sense_df: pd.DataFrame) -> pd.DataFrame:
    """Aggregate per-sense zero-shot rows to the five Paper 4 registers, with derived mixed."""
    if sense_df.empty:
        return pd.DataFrame()

    meta_cols = [c for c in ["term", "sense_id", "sense_size"] if c in sense_df.columns]
    score_cols = [c for c in sense_df.columns if c not in meta_cols]

    reg_scores = _aggregate_to_main_registers(
        sense_df[score_cols],
        term_labels=sense_df["term"] if "term" in sense_df.columns else None,
    )

    out = sense_df[meta_cols].copy()
    for col in reg_scores.columns:
        out[col] = reg_scores[col].to_numpy()

    return out

        return False

    clean_sense = _aggregate_sense_rows_to_registers(clean_sense)

    max_size = 0
    for sense_frame in (full_sense, clean_sense):
        if not sense_frame.empty and "sense_size" in sense_frame.columns:
            max_size = max(
                max_size,
                _to_int_scalar(pd.to_numeric(sense_frame["sense_size"], errors="coerce").fillna(0).max()),

    if max_size <= 0:
        max_size = 1

    sense_cmap = plt.get_cmap("Set2")

        sense_cmap(pos)
        for pos in np.linspace(0.0, 1.0, num=max(len(MAIN_REGISTER_ORDER), 8), endpoint=False)

        register_name: register_colors[reg_idx % len(register_colors)]

def build_master_results_csv() -> None:
    """
    Flatten every CSV under results_paper4/ into one long-form master CSV.

    Writes:
        results_paper4/master/master_results_long.csv
    """
    master_rows: List[Dict[str, Any]] = []

    for csv_path in sorted(RES.rglob("*.csv")):
        # Skip anything already inside the master folder or the Engawa problem folder
        if _is_under(csv_path, MASTER) or _is_under(csv_path, ENGAWA_PROBLEM):
            continue

        try:
            df = pd.read_csv(csv_path)
        except Exception as exc:
            LOGGER.warning("Could not read %s for master CSV: %s", csv_path, exc)
            continue

        if df.empty and len(df.columns) == 0:
            continue

        work = df.copy()

        # Preserve exported index columns from wide matrices
        if len(work.columns) > 0 and str(work.columns[0]).startswith("Unnamed:"):
            work = work.rename(columns={work.columns[0]: "__index__"})

        id_candidates = [
            "__index__",
            "model",
            "mode",
            "term",
            "label",
            "concept",
            "top_label",
            "other_label",
            "top_concept",
            "other_concept",
        ]
        id_cols = [c for c in id_candidates if c in work.columns]

        if not id_cols and len(work.columns) > 0:
            id_cols = [work.columns[0]]

        value_cols = [c for c in work.columns if c not in id_cols]
        if not value_cols and len(work.columns) == 1:
            value_cols = [work.columns[0]]
            id_cols = []

        for row_number, (row_idx, row) in enumerate(work.iterrows()):
            row_context = {
                c: (None if pd.isna(row[c]) else str(row[c]))
                for c in id_cols
                if c in work.columns
            }
            row_context_json = json.dumps(row_context, ensure_ascii=False)
            source_row_index = "" if pd.isna(row_idx) else str(row_idx)

            if value_cols:
                for metric in value_cols:
                    master_rows.append(
                        {
                            "source_file": _display_relpath(csv_path, RES),
                            "source_stem": csv_path.stem,
                            "row_number": row_number,
                            "source_row_index": source_row_index,
                            "row_context": row_context_json,
                            "metric": str(metric),
                            "value": row[metric],
                        }
                    )
            else:
                master_rows.append(
                    {
                        "source_file": _display_relpath(csv_path, RES),
                        "source_stem": csv_path.stem,
                        "row_number": row_number,
                        "source_row_index": source_row_index,
                        "row_context": row_context_json,
                        "metric": "",
                        "value": "",
                    }
                )

    out_path = MASTER / "master_results_long.csv"
    if master_rows:
        pd.DataFrame(master_rows).to_csv(out_path, index=False)
        LOGGER.info("Wrote master long-form results CSV to %s", out_path)
    else:
        LOGGER.info("No rows available for %s", out_path)


def _prepare_frame_for_auto_plot(csv_path: Path) -> Tuple[Optional[pd.DataFrame], str]:
    """
    Convert a results CSV into a numeric DataFrame suitable for plotting.

    Returns:
        (frame, kind)
        kind:
        - term_label_similarity
        - term_concept_similarity
        - k_metrics
        - generic_numeric
        - no_numeric_columns
        - empty
    """
    df = pd.read_csv(csv_path)

    if df.empty and len(df.columns) == 0:
        return None, "empty"

    work = df.copy()

    if len(work.columns) > 0 and str(work.columns[0]).startswith("Unnamed:"):
        work = work.rename(columns={work.columns[0]: "__index__"})

    if {"term", "label", "similarity"}.issubset(work.columns):
        frame = work.pivot_table(
            index="term",
            columns="label",
            values="similarity",
            aggfunc="mean",
        )
        return frame, "term_label_similarity"

    if {"term", "concept", "similarity"}.issubset(work.columns):
        frame = work.pivot_table(
            index="term",
            columns="concept",
            values="similarity",
            aggfunc="mean",
        )
        return frame, "term_concept_similarity"

    if "k" in work.columns:
        metric_cols = [
            c for c in work.columns
            if c != "k" and pd.api.types.is_numeric_dtype(work[c])
        ]
        if metric_cols:
            return work[["k"] + metric_cols].copy(), "k_metrics"

    index_col: Optional[str] = None
    for candidate in ("__index__", "term", "label", "concept", "model"):
        if candidate in work.columns and work[candidate].is_unique:
            index_col = candidate
            break

    numeric_cols = [c for c in work.columns if pd.api.types.is_numeric_dtype(work[c])]
    if index_col is not None:
        numeric_cols = [c for c in numeric_cols if c != index_col]

    if not numeric_cols:
        return None, "no_numeric_columns"

    frame = work[[index_col] + numeric_cols].copy() if index_col else work[numeric_cols].copy()
    if index_col:
        frame = frame.set_index(index_col)

    return frame, "generic_numeric"


def _plot_one_results_csv(csv_path: Path, plot_dir: Path) -> Tuple[str, str, str]:
    """
    Plot one results CSV.

    Returns:
        (status, detail, plot_filename)
    """
    frame, kind = _prepare_frame_for_auto_plot(csv_path)
    if frame is None or frame.empty:
        return "skipped", kind, ""

    rel_stem = _display_relpath(csv_path.with_suffix(""), RES)
    plot_filename = f"{_safe_slug(rel_stem)}.png"
    out_path = plot_dir / plot_filename

    if kind == "k_metrics":
        plt.figure(figsize=(10, 6))
        k_vals = frame["k"].astype(float)
        metric_cols = [c for c in frame.columns if c != "k"]
        for col in metric_cols:
            plt.plot(k_vals, frame[col].astype(float), marker="o", label=str(col))
        plt.xlabel("k")
        plt.ylabel("value")
        plt.title(csv_path.stem)
        plt.legend(fontsize=8)
        plt.tight_layout()
        _save_png_svg(out_path, dpi=160)
        plt.close()
        return "plotted", kind, plot_filename

    if frame.shape[0] == 1 or frame.shape[1] == 1:
        plt.figure(figsize=(10, 6))
        if frame.shape[0] == 1:
            ser = frame.iloc[0].dropna().astype(float)
        else:
            ser = frame.iloc[:, 0].dropna().astype(float)

        if ser.empty:
            plt.close()
            return "skipped", "empty_vector", ""

        ser.plot(kind="bar")
        plt.title(csv_path.stem)
        plt.xticks(rotation=90, fontsize=8)
        plt.tight_layout()
        _save_png_svg(out_path, dpi=160)
        plt.close()
        return "plotted", kind, plot_filename

    plt.figure(
        figsize=(
            max(10.0, float(frame.shape[1]) * 0.45),
            max(4.5, float(frame.shape[0]) * 0.30),
        )
    )
    sns.heatmap(
        frame.astype(float),
        cmap="coolwarm",
        annot=(frame.shape[0] <= 30 and frame.shape[1] <= 30),
        fmt=".2f",
        annot_kws={"fontsize": 7},
        cbar_kws={"label": "value"},
    )
    plt.title(csv_path.stem)
    plt.xticks(rotation=90, fontsize=8)
    plt.yticks(fontsize=8)
    plt.tight_layout()
    _save_png_svg(out_path, dpi=160)
    plt.close()
    return "plotted", kind, plot_filename

def build_master_figures_pdf() -> None:

def main() -> None:

    if RUN_POLYSEMY_FULL:
        with Timer("Contextual cluster recovery stage (FULL)"):

        with Timer("Contextual cluster recovery stage (CLEAN)"):
            run_polysemy_for_mode("CLEAN", TARGETS)

    scores_by_model_clean: Dict[str, pd.DataFrame] = {}

    flags_by_model_clean: Dict[str, pd.DataFrame] = {}
    cons_flags: Optional[pd.DataFrame] = None
    sa_flags_main: Optional[pd.DataFrame] = None

    if RUN_ZERO_SHOT:
        banner("ARCHITECTURE-FACING REGISTER ALIGNMENT — zero-shot model space (FULL + CLEAN)")
        with Timer("Architecture-facing register alignment stage"):

            # ---------- FULL ----------
            try:
                vecs_by_model_full = per_model_vectors(TARGETS, CATEGORY_LABELS, mode="FULL")
            except RuntimeError as exc:
                LOGGER.error("Zero-shot (FULL) model loading failed: %s", exc)
                vecs_by_model_full = {}

            for model_name, block in vecs_by_model_full.items():
                try:
                    sim_df = similarity_table(block, TARGETS, CATEGORY_LABELS)
                    scores_by_model_full[model_name] = sim_df
                    sim_df.to_csv(str(RES / f"zs_similarity_{model_name}_full.csv"))
                    flag_df = to_flags(sim_df)
                    flags_by_model_full[model_name] = flag_df
                    flag_df.to_csv(str(RES / f"zs_flags_{model_name}_full.csv"))
                except (ValueError, RuntimeError) as exc:
                    LOGGER.warning("Zero-shot (FULL) failed for %s: %s", model_name, exc)

            if scores_by_model_full:

                # Canonical FULL filenames for downstream dashboard/report logic

                cons_flags = to_flags(cons_df_full)
                cons_flags.to_csv(str(RES / "zs_flags_consensus_full.csv"))

                        flags_by_model_full["bert_base"],
                        flags_by_model_full["bert_ft"],
                    )

                        str(RES / "zs_jaccard_base_vs_ft_full.csv"),
                        index=False,
                    )

                cons_plot = cons_df_full.dropna(how="all").copy()
                cons_pct = rowwise_percent_scale(cons_plot).round(2)
                cons_pct.to_csv(str(RES / "zs_similarity_consensus_full_pct.csv"))

                if cons_pct.shape[0] >= 2 and cons_pct.shape[1] >= 2:
                    row_dist = pdist(cons_pct.to_numpy(dtype=float), metric="cosine")
                    row_link = sch.linkage(row_dist, method="average")
                    row_order = sch.leaves_list(row_link)
                    cons_pct = cons_pct.iloc[row_order]

                    col_dist = pdist(cons_pct.to_numpy(dtype=float).T, metric="cosine")
                    col_link = sch.linkage(col_dist, method="average")
                    col_order = sch.leaves_list(col_link)
                    cons_pct = cons_pct.iloc[:, col_order]

                short_xticks_full = [CATEGORY_LABEL_SHORT.get(lbl, lbl) for lbl in cons_pct.columns]

                plt.figure(
                    figsize=(
                        max(12.0, float(len(cons_pct.columns)) * 0.5),
                        max(6.0, float(len(cons_pct.index)) * 0.4),
                    )
                )
                ax = sns.heatmap(
                    cons_pct,
                    cmap="coolwarm",
                    vmin=0.0,
                    vmax=100.0,
                    annot=True,
                    fmt=".0f",
                    annot_kws={"fontsize": 8},
                    cbar_kws={"label": "row-wise min-max scaled contrast (0=lowest, 100=highest)"},
                )
                ax.set_xticklabels(short_xticks_full, rotation=90, fontsize=9)
                ax.set_yticklabels(cons_pct.index, fontsize=9)

                plt.title("Architecture-facing register alignment (FULL, row-wise min-max scaled contrast)")
                plt.tight_layout()
                _save_png_svg(FIG / "zs_similarity_consensus_full.png", dpi=160)
                plt.close()

            # ---------- CLEAN ----------
            try:
                vecs_by_model_clean = per_model_vectors(TARGETS, CATEGORY_LABELS, mode="CLEAN")
            except RuntimeError as exc:
                LOGGER.error("Zero-shot (CLEAN) model loading failed: %s", exc)
                vecs_by_model_clean = {}

            for model_name, block in vecs_by_model_clean.items():
                try:
                    sim_df = similarity_table(block, TARGETS, CATEGORY_LABELS)
